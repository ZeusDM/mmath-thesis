\chapter{Bijective combinatorics for Macdonald polynomials} \label{chap:bijective}

In this chapter, we describe \cite{KLO22}'s bijection between
\begin{equation*}
    \Fill[\mathcal{A}](\mu) \quad \text{and} \quad \Fill[\mathcal{A}](s_i \permact \mu),
\end{equation*}
which preserves the statistics $\inv$ and $\maj$.

The existence of this bijection implies that 
\begin{equation}
    \tilde{H}_\mu[X;q,t] = \tilde{H}_{s_i \permact \mu}[X;q,t],
\end{equation}
and by applying this result repeatedly, we obtain
\begin{equation}
    \tilde{H}_{\mu}[X;q,t] = \tilde{H}_{\alpha}[X;q,t],
\end{equation}
where $\alpha$ is a rearrangement of the partition $\mu$.

\section{The bijection}

In this section, we define the bijection \(\phi \colon \Fill(\alpha) \to \Fill(s_i \permact \alpha)\) from \cite{KLO22} that preserves the statistics \(\inv\) and \(\maj\).
As an auxiliary construction,
we also define another bijection \(\psi \colon \Fill(\alpha) \to \Fill(s_i \permact \alpha)\) that changes the statistics \(\inv\) and \(\maj\) in a controlled way.
The manner in which \(\psi\) changes the statistics is specified in \Cref{thm:psi-maj-inv}.
This auxiliary bijection \(\psi\) will also be important in a later section.
Without loss of generality, we restrict to the case \(\alpha_{i} > \alpha_{i+1}\).

Both of \(\phi\) and \(\psi\) follow a very rigid structure.
If a filling \(T \in \Fill(\alpha)\) is mapped to a filling \(U \in \Fill(s_i \permact \alpha)\) by one of these maps, then
\begin{itemize}
    \item \(T(j, r) = U(j, r)\) for all \(j \in \interval{n} \setminus \{i, i+1\}\) and all \(r \in \interval{\alpha_j}\),
    that is, all columns except \(i\) and \(i+1\) are identical in \(T\) and \(U\);
    \item \(T(i, r) = U(i+1, r)\) for all \(r \in \interval{\alpha_{i+1} + 1, \alpha_i}\),
    that is, the topmost \(\alpha_i - \alpha_{i+1}\) entries in column \(i\) of \(T\) are copied to the topmost \(\alpha_i - \alpha_{i+1}\) entries in column \(i+1\) of \(U\);
    \item \(\{ T(i, r), T(i+1, r) \} = \{ U(i, r), U(i+1, r) \}\) for all \(r \in \interval{\alpha_{i+1}}\),
    that is, the entries in rows \(1, \dots, \alpha_{i+1}\) of columns \(i\) and \(i+1\) in \(T\) and \(U\) are the same, but they might be in the same positions or in swapped positions.
\end{itemize}

Given these restrictions, the only freedom we have in defining \(\phi\) and \(\psi\) is to decide, for each \(r \in \interval{\alpha_{i+1}}\), whether to set
\begin{equation*}
    \big(U(i,r), U(i+1,r)\big) = \big(T(i,r), T(i+1,r)\big) \text{ or } \big(U(i,r), U(i+1,r)\big) = \big(T(i+1,r), T(i,r)\big).
\end{equation*}
In the first case, we say the entries of \(T\) and \(U\) in row \(r\) of columns \(i\) and \(i+1\) are \vocab{not swapped};
in the second case, we say they are \vocab{swapped}.

\subsection{Iterative algorithm.}
Let \(T \in \Fill(\alpha)\) with \(\alpha_i > \alpha_{i+1}\).
We describe the algorithm to compute \(U = \phi(T)\) or \(U = \psi(T)\).
The algorithm uses a \vocab{coin} with two sides, \(\coinphi\) and \(\coinpsi\), as a bookkeeping device.
Each row from \(1\) to \(\alpha_{i+1} - 1\) is, at some point of the algorithm, assigned a \vocab{coin value} in \(\{\coinphi, \coinpsi\}\).
Initially, assign the coin value of row \(1\) to be \(\coinphi\) if computing \(U = \phi(T)\), or \(\coinpsi\) if computing \(U = \psi(T)\).

In addition to the coin value, each row from \(1\) to \(\alpha_{i+1}\) is labeled from \(\{\labelN, \labelS, \labelC\}\).
A label of \(\labelN\) means the entries in row \(r\) of columns \(i\) and \(i+1\) in \(T\) and \(U\) are \emph{not} swapped;
a label of \(\labelS\) means they are \emph{swapped};
and a label of \(\labelC\) indicates that the behavior for row \(r\) should \emph{copy} the behavior for the row above it.

We process the rows iteratively from row \(1\) up to row \(\alpha_i\).
For each \(r \in \{1,2,\dots,\alpha_{i+1}-1\}\),
let \( a = T(i,r+1)\), \(b = T(i+1,r+1)\), \(c = T(i,r)\), and \(d = T(i+1,r)\).
Note that \(\ind{a, d} = \ind{b, c} \neq \ind{a, c} = \ind{b, d}\) is impossible by transitivity,
thus \( \ind{a,d} + \ind{b,c} - \ind{a,c} - \ind{b,d} \in \{1, 0, -1\} \).

\begin{description}
    \item[Case I.] If 
    \(\ind{a,d} + \ind{b,c} - \ind{a,c} - \ind{b,d} = \pm 1\),
    (cases 3,4,9,10 in \cite{KLO22}),  
    then assign row \(r\) the label \(\labelC\).
    Assign row \(r+1\) the coin value of row \(r\) if \(\chi_{ab} = \chi_{cd}\) (cases 3,9),
    and assign row \(r+1\) the opposite coin value of row \(r\) if \(\chi_{ab} \neq \chi_{cd}\) (cases 4,10).

    \item[Case II.] If
    \(\ind{a,c} = \ind{a,d} = \ind{b,c} = \ind{b,d}\)
    (cases 1,2,11,12),
    then label row \(r\) with \(\labelN\) (resp.~\(\labelS\)) if its coin value is \(\coinphi\) (resp.~\(\coinpsi\)).
    Assign row \(r+1\) the coin value \(\coinphi\).
    
    \item[Case III.] If 
    \(\ind{a,c} = \ind{b,c} \neq \ind{a,d} = \ind{b,d}\)  
    (cases 7,8),  
    then label row \(r\) with \(\labelS\) (resp.~\(\labelN\)) if its coin value is \(\coinphi\) (resp.~\(\coinpsi\)).
    Assign row \(r+1\) the coin value \(\coinphi\).
    
    \item[Case IV.] If
    \(\ind{a,c} = \ind{a,d} \neq \ind{b,c} = \ind{b,d}\)
    (cases 5,6),
    then label row \(r\) with \(\labelN\) (resp.~\(\labelS\)) if its coin value is \(\coinphi\) (resp.~\(\coinpsi\)).
    Assign row \(r+1\) the coin value \(\coinpsi\).
\end{description}

Finally, we process row \(r = \alpha_{i+1}\) (the last row of column \(i+1\)) as follows.
Let \( a = T(i,r+1)\), \(c = T(i,r)\), and \(d = T(i+1,r)\).

\begin{description}
    \item[Base I.] If \(\ind{a,c} \neq \ind{a,d}\), then label row \(r\) with
    \(\labelS\) (resp.~\(\labelN\)) if its coin value is \(\coinphi\) (resp.~\(\coinpsi\)).

    \item[Base II.] If \(\ind{a,c} = \ind{a,d}\), then label row \(r\) with
    \(\labelN\) (resp.~\(\labelS\)) if its coin value is \(\coinphi\) (resp.~\(\coinpsi\)).
\end{description}

Finally, build \(U\) row by row:
\begin{itemize}
    \item If row \(r\) is labeled \(\labelN\), set
    \(U(i,r) = T(i,r)\) and \(U(i+1,r) = T(i+1,r)\).
    \item If row \(r\) is labeled \(\labelS\), set
    \(U(i,r) = T(i+1,r)\) and \(U(i+1,r) = T(i,r)\).
    \item If row \(r\) is labeled \(\labelC\), then use the same rule as the nearest row above it labeled \(\labelN\) or \(\labelS\).
    Note that since row \(\alpha_{i+1}\) is always labeled \(\labelN\) or \(\labelS\), there is always a nearest row above it labeled \(\labelN\) or \(\labelS\).
\end{itemize}

\subsection{Recursive algorithm.}
The algorithm above may be described recursively in terms of the labels, instead of iteratively.

Let \(T \in \Fill(\alpha)\) with \(\alpha_i > \alpha_{i+1}\).
Assign the coin value of row \(1\) to be \(\coinphi\) if computing \(U = \phi(T)\), or \(\coinpsi\) if computing \(U = \psi(T)\).
If column \(i+1\) consists of a single entry, then assign the label of this row according to Base~I or~II.
Otherwise, assign the label of row \(1\) according one of Cases~I--IV,
and at the same time determine the coin value to be used for the next row.
Remove the first row to obtain the sub-filling \(T'\), and apply the algorithm recursively to \(T'\) with this new coin value, obtaining the labels for rows \(2,\dots,\alpha_{i+1}\).
Once all labels are determined, construct \(U\) in a final pass by applying the rules for \(\labelN,\labelS,\labelC\) as before.

It is straightforward to verify that both the iterative and recursive descriptions of the algorithm are equivalent.
This recursive interpretation is particularly useful for proving properties about the bijections \(\phi\) and \(\psi\), as it allows us to use induction on the number of rows in the filling.

\section{Properties of the bijections \texorpdfstring{\(\phi\)}{𝜙} and \texorpdfstring{\(\psi\)}{𝜓}}

\begin{proposition}
    The maps \(\phi\) and \(\psi\) are bijections.
\end{proposition}

\begin{theorem}[\cite{KLO22}] \label{thm:phi-maj-inv}
    The bijection \(\phi\) preserves the statistics \(\inv\) and \(\maj\); that is, for all \(T \in \Fill(\alpha)\), if \(U = \phi(T)\), then
    \begin{equation*}
        \inv(T) = \inv(U) \quad \text{and} \quad \maj(T) = \maj(U).
    \end{equation*}
\end{theorem}

\begin{theorem}[\cite{KLO22}] \label{thm:psi-maj-inv}
    The bijection \(\psi\) modifies the statistics \(\inv\) and \(\maj\) in a specific way. For any \(T \in \Fill(\alpha)\) and \(U = \psi(T)\), we have
    \begin{itemize}
        \item if \(\ind{{T(i,1)},{T(i+1,1)}} = 1\) and \(\ind{{U(i,1)},{U(i+1,1)}} = 1\), then
        \begin{equation*}
            \maj(T) = \maj(U) - (L+1)
            \quad \text{and} \quad
            \inv(T) = \inv(U) + A,
        \end{equation*}
        \item if \(\ind{{T(i,1)},{T(i+1,1)}} = 1\) and \(\ind{{U(i,1)},{U(i+1,1)}} = 0\), then
        \begin{equation*}
            \maj(T) = \maj(U)
            \quad \text{and} \quad
            \inv(T) = \inv(U) + 1,
        \end{equation*}
        \item if \(\ind{{T(i,1)},{T(i+1,1)}} = 0\) and \(\ind{{U(i,1)},{U(i+1,1)}} = 1\), then
        \begin{equation*}
            \maj(T) = \maj(U)
            \quad \text{and} \quad
            \inv(T) = \inv(U) - 1,
        \end{equation*}
        \item and if \(\ind{{T(i,1)},{T(i+1,1)}} = 0\) and \(\ind{{U(i,1)},{U(i+1,1)}} = 0\), then
        \begin{equation*}
            \maj(T) = \maj(U) + (L+1)
            \quad \text{and} \quad
            \inv(T) = \inv(U) - A,
        \end{equation*}
    \end{itemize}
    where \(L = \leg(...)\) and \(A = \arm(...)\).
\end{theorem}
