\chapter{Bijective combinatorics for Macdonald polynomials} \label{chap:bijective}

This chapter is split into two sections. \Cref{sec:bijective-non-augmented} provides bijections on non-augmented fillings to obtain formulas for the modified Macdonald polynomials \(\tilde{H}_\mu(\mathbf{x}; q, t)\).
\Cref{sec:bijective-augmented} provides bijections on augmented fillings to obtain formulas for the permuted-basement Macdonald polynomials \(E_\alpha^\sigma(\mathbf{x}; q, t)\).

In \Cref{sec:bijective-non-augmented},
we study a rearrangement property of the modified Macdonald polynomials \(\tilde{H}_\mu(\mathbf{x}; q, t)\),
\Cref{thm:tildeH-rearrangement},
which states that, for any partition \(\mu\) and any rearrangement \(\alpha\) of \(\mu\),
\begin{equation} \label{eq:tildeH-rearrangement}
    \sum_{T \in \Fill[\positives](\mu)} q^{\maj(T)} t^{\inv(T)} \mathbf{x}^T
    =
    \sum_{T \in \Fill[\positives](\alpha)} q^{\maj(T)} t^{\inv(T)} \mathbf{x}^T.
\end{equation}
The left-hand sum over \(\Fill[\positives](\mu)\) is the celebrated combinatorial expression for \(\tilde{H}_{\mu}(\mathbf{x}; q, t)\) from \cite{HHL05} (see \Cref{subsec:modified-macdonald}), where they show that the sum satisfies the conditions of \Cref{prop:modified-macdonald-characterization}.
In \cite{HHL08}, the rearrangement property was established by proving that the right-hand sum over \(\Fill[\positives](\alpha)\) is also equal to \(\tilde{H}_{\mu}(\mathbf{x}; q, t)\),
by showing that it also satisfies the conditions of \Cref{prop:modified-macdonald-characterization}.

We present a bijective proof of \eqref{eq:tildeH-rearrangement}, without going through the algebraic characterization of
\(\tilde{H}_\mu(\mathbf{x}; q, t)\) from \Cref{prop:modified-macdonald-characterization}.
The key step is \Cref{thm:phi-maj-inv},
which guarantees the existence of a bijection \(\phi \colon \Fill(\alpha) \to \Fill(s_i \permact \alpha)\) that preserves the content, the inversion number, and the major index.

There are two equivalent constructions of such bijection \(\phi\).
The first construction partitions the filling into blocks and decides how to transform each block based on local conditions.
The second construction is an iterative row-by-row algorithm from \cite{KLO22}, which we reformulate in a clearer way than the original presentation.
Each construction has its advantages, and we include both for completeness.
We prove that both algorithms coincide in \Cref{subsec:equivalence-of-definitions},
and then prove that \(\phi\) satisfies the desired properties in \Cref{subsec:properties-of-bijections} using the first construction.

An auxiliary bijection \(\psi \colon \Fill[\mathcal{A}](\alpha) \to \Fill[\mathcal{A}](s_i \permact \alpha)\) is also defined, that preserves \(\content\) but changes \(\inv\) and \(\maj\) in a controlled way, as specified in \Cref{thm:psi-maj-inv}.

In \Cref{sec:bijective-augmented},
we prove \Cref{theorem:three-term}, which gives an identity for permuted-basement Macdonald polynomials of the form
\begin{equation} \label{eq:three-term-intro}
    E_\alpha^\sigma(\mathbf{x}; q, t)
    =
    E_{s_i \permact \alpha}^{\sigma s_i}(\mathbf{x}; q, t)
    +
    c_{\alpha, \sigma, i}(q, t) E_{s_i \permact \alpha}^\sigma(\mathbf{x}; q, t),
\end{equation}
whenever \(\alpha_i > \alpha_{i+1}\), where \(c_{\alpha, \sigma, i}(q, t)\) is an explicit rational function in \(q\) and \(t\).

We present a bijective proof of \eqref{eq:three-term-intro}.
We use the bijection \(\phi\) on non-augmented fillings from \Cref{sec:bijective-non-augmented} and an auxiliary bijection \(\psi\) as a building blocks for a bijection on augmented fillings that implies \eqref{eq:three-term-intro}.

\section{Bijections on non-augmented fillings and the rearrangement property of \texorpdfstring{\(\tilde{H}_\mu\)}{H̃μ}} \label{sec:bijective-non-augmented}

In this section, we define the bijection \(\phi \colon \Fill(\alpha) \to \Fill(s_i \permact \alpha)\) from \cite{KLO22} that preserves the statistics \(\inv\) and \(\maj\).
As an auxiliary construction,
we also define another bijection \(\psi \colon \Fill(\alpha) \to \Fill(s_i \permact \alpha)\) that changes the statistics \(\inv\) and \(\maj\) in a controlled way, as specified in \Cref{thm:psi-maj-inv}.
This auxiliary bijection \(\psi\) is also important in \Cref{sec:bijective-augmented}.
% Without loss of generality, we restrict to the case \(\alpha_{i} > \alpha_{i+1}\), and use the inverse bijections to handle the case \(\alpha_{i} < \alpha_{i+1}\).
% \todo{Can I remove this ``without loss of generality''?}

We assume \(\alpha_i \neq \alpha_{i+1}\) throughout this section, since the case \(\alpha_i = \alpha_{i+1}\) has \(\alpha = s_i \permact \alpha\) and the identity map serves as a suitable choice for \(\phi\).

Both of \(\phi\) and \(\psi\) follow a very rigid structure.
For a non-augmented filling \(T \in \Fill(\alpha)\) mapped
to a filling \(U \in \Fill(s_i \permact \alpha)\) by one of these maps, then:
all columns except \(i\) and \(i+1\) are identical in \(T\) and \(U\), that is,
\begin{equation} \label{eq:restriction-other-columns}
    T(j, r) = U(j, r)
    \quad\text{for all } j \in \interval{n} \setminus \{i, i+1\}
    \text{ and all } r \in \interval{\alpha_j};
\end{equation}
and the entries in each row \(r \in \interval{\max(\alpha_i, \alpha_{i+1})}\) of columns \(i\) and \(i+1\) in \(T\) and \(U\) are the same, that is,
\begin{equation} \label{eq:restriction-swap-per-row}
    \{ T(i, r), T(i+1, r) \}
    =
    \{ U(i, r), U(i+1, r) \}
    \quad\text{for all } r \in \interval{\max(\alpha_i, \alpha_{i+1})}.
\end{equation}
We refer to these as the \vocab{local restrictions}.

When \(r > \min(\alpha_i, \alpha_{i+1})\) in \eqref{eq:restriction-swap-per-row}, exactly one of the two entries involved is undefined in each filling, so the condition should be interpreted as stating that the defined entry in \(T\) is copied to the corresponding defined position in \(U\).

By requiring that \(\phi\) and \(\psi\) satisfy these local restrictions, the only remaining freedom in determining \(\phi\) and \(\psi\) is to decide, for each \(r \in \interval{\min(\alpha_i, \alpha_{i+1})}\), whether to set
\begin{equation*}
    \begin{aligned}
        \big(U(i,r), U(i+1,r)\big) &= \big(T(i,r), T(i+1,r)\big) \text{ or }\\
        \big(U(i,r), U(i+1,r)\big) &= \big(T(i+1,r), T(i,r)\big).
    \end{aligned}
\end{equation*}
In the first case, we say the entries of \(T\) and \(U\) in row \(r\) of columns \(i\) and \(i+1\) are \vocab{not swapped};
in the second case, we say they are \vocab{swapped}.
Below, we present two different algorithms to decide, for each \(r \in \interval{\min(\alpha_i, \alpha_{i+1})}\), whether the entries in row \(r\) of columns \(i\) and \(i+1\) are swapped or not swapped.
See \Cref{fig:local-restrictions} for an illustration of the local restrictions.

\subsection{Block structure algorithm} \label{subsec:block-structure}

Let \(h = \min(\alpha_i, \alpha_{i+1})\).
For the block structure algorithm,
we split the rows \(\interval{h}\) into intervals, called \emph{blocks},
and then decide, for each block, whether to swap all its rows or not swap all its rows.

To define the blocks, we define a criterion for when two consecutive rows are grouped together.
Since our goal is to construct bijections that preserve the inversion statistic \(\inv\) and inversion triples occur between two consecutive rows,
the condition we use is based on the behavior of \(\ind{{}}\) on four elements.
Before proceeding, we prove some properties of \(\ind{{}}\) when applied to four elements \(a, b, c, d \in \mathcal{A}\).

\begin{proposition} \label{prop:ind-magic}
    Let \(a, b, c, d \in \mathcal{A}\) be four elements of an alphabet \(\mathcal{A}\).
    Then,
    \begin{equation}
        \ind{a, d} + \ind{b, c} - \ind{a, c} - \ind{b, d} \in \{-1, 0, 1\}.
    \end{equation}
\end{proposition}

\begin{proof}
    Since \(\ind{x, y} \in \{0, 1\}\) for all \(x, y \in \mathcal{A}\), it's immediate that the expression only take values in \(\{-2, -1, 0, 1, 2\}\).
    For its absolute value to be \(2\), we need \(\ind{a, d} = \ind{b, c} = 1 - \ind{a, c} = 1 - \ind{b, d}\).
    This implies that \(a \geq d \geq b \geq c \geq a\) or \(a \leq d \leq b \leq c \leq a\); either way, we get \(a = b = c = d\), for which the expression evaluates to \(0\), a contradiction.
    Thus, the expression only take values in \(\{-1, 0, 1\}\).
\end{proof}

For distinct elements \(a, b, c, d \in \mathcal{A}\), the value of \(\ind{a, d} + \ind{b, c} - \ind{a, c} - \ind{b, d}\) equals \(0\) if and only if, when placing the four elements on a line in increasing order and drawing arcs between \(a\) and \(b\), and between \(c\) and \(d\), the arcs do not cross.
See \Cref{fig:ind-magic} for two examples.
We have \Cref{prop:even-ind-magic,prop:odd-ind-magic} that give some other equivalent characterizations of whether the expression is \(0\) or \(\pm 1\), which work even when the elements are not distinct.

\begin{proposition} \label{prop:even-ind-magic}
    Let \(a, b, c, d \in \mathcal{A}\).
    The following statements are equivalent:
    \begin{enumerate}
        \item \label{item:even-ind}
        \(\ind{a, d} + \ind{b, c} - \ind{a, c} - \ind{b, d} = 0\);
        \item \label{item:ind-abx}
        \(\ind{a, b, c} = \ind{a, b, d}\);
        \item \label{item:ind-xcd}
        \(\ind{a, c, d} = \ind{b, c, d}\).
    \end{enumerate}
\end{proposition}

\begin{proof}
    Expanding the definition of \(\ind{x, y, z}\), \Cref{item:ind-abx} is equivalent to
    \(\ind{a, b} + \ind{b, c} - \ind{a, c} = \ind{a, b} + \ind{b, d} - \ind{a, d}\),
    which rearranges to \Cref{item:even-ind}.
    The equivalence between \Cref{item:ind-xcd} and \Cref{item:even-ind} is analogous.
\end{proof}

\begin{proposition} \label{prop:odd-ind-magic}
    Let \(a, b, c, d \in \mathcal{A}\).
    The following statements are equivalent to the negation of those in \Cref{prop:even-ind-magic}:
    \begin{enumerate}[start=4]
        \item \label{item:odd-ind}
        \(\ind{a, d} + \ind{b, c} - \ind{a, c} - \ind{b, d} = \pm 1\);
        \item \label{item:odd-ind-2}
        \(\ind{a, d} + \ind{b, c} + \ind{a, c} + \ind{b, d}\) is odd;
        \item \label{item:triple-equal-ind}
        \(\ind{a, b, c} = \ind{b, c, d} = \ind{a, d, c}\) and \(a \neq b\) and \(c \neq d\);
    \end{enumerate}
\end{proposition}

\begin{proof}
    The equivalence between \Cref{item:odd-ind} and the negation of \Cref{item:even-ind} is immediate from \Cref{prop:ind-magic}, as well as the equivalence between \Cref{item:odd-ind} and \Cref{item:odd-ind-2} using modulo \(2\).
    Computational verification shows that \Cref{item:odd-ind} and \Cref{item:triple-equal-ind}
    are satisfied for exactly the same relative configurations of \(a, b, c, d\), namely one of the following \(40\) configurations:

    c > b > a \overset{-}{=} d, \quad

Let \(r \in \interval{h - 1}\).
Let \( a = T(i,r+1)\), \(b = T(i+1,r+1)\), \(c = T(i,r)\), and \(d = T(i+1,r)\).
We say that rows \(r\) and \(r + 1\) of \(T\) are \vocab{not intertwined} if \(a, b, c, d\) satisfy the equivalent conditions of \Cref{prop:even-ind-magic}, and \vocab{intertwined} if they satisfy the equivalent conditions of \Cref{prop:odd-ind-magic}.
Note that the condition is invariant under swapping \(a\) with \(b\), as well as invariant under swapping \(c\) with \(d\).

The interval of rows \(\interval{h}\) is uniquely partitioned into intervals, called \vocab{blocks}, such that two consecutive rows in the same block are intertwined, and two consecutive rows in different blocks are not intertwined.
The partition of \(\interval{h}\) into blocks is called the \vocab{block structure \textup{(}of columns \(i\) and \(i+1\)\textup{)} of the filling \(T\)}.
\Cref{fig:blocks} shows a filling and its blocks.

Note that, if the two entries of row \(r\) of columns \(i\) and \(i+1\) in \(T\) are equal, then rows \(r - 1\) and \(r\) are not intertwined, and rows \(r\) and \(r + 1\) are not intertwined (if these rows exist).
Therefore, such row \(r\) forms a block of size \(1\) by itself, and swapping or not swapping it has the same effect.

Let's interpret the intertwining condition via~\Cref{item:triple-equal-ind} of \Cref{prop:odd-ind-magic}.
The number \(\ind{a, b, c}\) is the indicator of whether the triple \((i, r+1), (i+1, r+1), (i, r)\) is a Type A inversion triple in \(T\).
The number \(\ind{b, c, d}\) is the indicator of whether the triple \((i+1, r+1), (i, r), (i+1, r)\) is a Type B inversion triple in \(U\) if rows \(r\) and \(r + 1\) are not swapped.
The number \(\ind{a, d, c}\) is the indicator of whether the triple \((i, r+1), (i+1, r), (i, r)\) is a Type B inversion triple in \(U\) if rows \(r\) and \(r + 1\) are swapped.

Therefore,
if we require that rows in the same block are either all swapped or all not swapped in \(U\),
then we guarantee that the number of inversion triples within each block is the same in \(T\) and \(U\), regardless of whether the block is swapped or not swapped.

The remaining freedom is to decide, for each block, whether to swap it or not swap it.
We use this freedom to guarantee that the number of inversion triples between different blocks is also the same in \(T\) and \(U\).

Consider a block \([r_1, r_2]\) of \(T\).
Let
\begin{align*}
    e_1 &= T(i, r_2 + 1),  & e_2 &= T(i+1, r_2 + 1), \\
    a   &= T(i, r_2),      & b   &= T(i+1, r_2),     \\
    c   &= T(i, r_1),      & d   &= T(i+1, r_1),     \\
    f_1 &= T(i, r_1 - 1), \text{ and} & f_2 &= T(i+1, r_1 - 1).
\end{align*}
Edge cases are when the block is the bottommost one, i.e. \(r_1 = 1\),
in which case \(f_1 = f_2 = \infty\) by convention;
and when the block is the topmost one, i.e. \(r_2 = h\), in which case one of \(e_1, e_2\) is undefined.
For any well-defined \(e \in \{e_1, e_2\}\) and \(f \in \{f_1, f_2\}\),
we say that \((a, b, c, d, e, f)\) are \vocab{signature entries} of the block.
See \Cref{fig:general-block} for an illustration of the entries \(a, b, c, d, e_1, e_2, f_1, f_2\).

If the block is not the bottommost one, then rows \(r_1 - 1\) and \(r_1\) are not intertwined.
Therefore, \(c, d, f_1, f_2\) satisfy the equivalent conditions of \Cref{prop:even-ind-magic}, in particular, \(\ind{c, d, f_1} = \ind{c, d, f_2}\).
If the block is the bottommost one,
then we have \(\ind{c, d, f_1} = \ind{c, d, f_2} = \ind{c, d}\) by convention (as expected from \(f_1 = f_2 = \infty\)).
Therefore, in all cases, the value \(\ind{c, d, f}\) doesn't depend on the choice of \(f \in \{f_1, f_2\}\).

If the block is not the topmost one, then rows \(r_2\) and \(r_2 + 1\) are not intertwined.
Therefore, \(a, b, e_1, e_2\) satisfy the equivalent conditions of \Cref{prop:even-ind-magic}, in particular, \(\ind{e_1, a, b} = \ind{e_2, a, b}\).
If the block is the topmost one, then exactly one of \(e_1\) and \(e_2\) is defined, so there is no ambiguity in choosing \(e\) and in computing \(\ind{e, a, b}\).
Therefore, in all cases, the value \(\ind{e, a, b}\) doesn't depend on the choice of \(e \in \{e_1, e_2\}\) (as long as \(e\) is defined).

In all cases, the values \(\ind{c, d, f}\) and \(\ind{e, a, b}\) are uniquely defined for a block, independently of the choice of signature entries \((a, b, c, d, e, f)\).

\begin{definition}[The bijection \(\phi\) via blocks] \label{def:phi-via-blocks}
    Let \(T \in \Fill(\alpha)\).
    We define \(U = \phi(T) \in \Fill(s_i \permact \alpha)\)
    by requiring the local restrictions (i.e., \eqref{eq:restriction-other-columns} and \eqref{eq:restriction-swap-per-row})
    and requiring that, for each block of \(T\) with signature entries \((a, b, c, d, e, f)\),
    the rows of the block are swapped in \(U\) if and only if
    \begin{equation}
        \ind{c, d, f} \neq \ind{e, a, b}.
    \end{equation}
\end{definition}

\begin{definition}[The bijection \(\psi\) via blocks] \label{def:psi-via-blocks}
    Let \(T \in \Fill(\alpha)\).
    We define \(U = \psi(T) \in \Fill(s_i \permact \alpha)\)
    by requiring the local restrictions (i.e., \eqref{eq:restriction-other-columns} and \eqref{eq:restriction-swap-per-row})
    and requiring that, for each block of \(T\) with signature entries \((a, b, c, d, e, f)\),
    the rows of the block are swapped in \(U\) if and only if
    \begin{equation}
        \ind{c, d, f} \neq \ind{e, a, b} \quad \mathbin{\mathrm{xor}} \quad \text{(the block is the bottommost block)},
    \end{equation}
    where \(\mathbin{\mathrm{xor}}\) denotes the exclusive or operation, 
    whose result is true if and only if an odd number of its operands are true.
\end{definition}

To make things explicitly, both \(\phi\) and \(\psi\) swap a non-bottommost block if and only if \(\ind{c, d, f} \neq \ind{e, a, b}\).
The difference is in the bottommost block, which \(\phi\) swaps if and only if \(\ind{c, d} \neq \ind{e, a, b}\),
while \(\psi\) swaps it if and only if \(\ind{c, d} = \ind{e, a, b}\).

In \Cref{prop:phi-inverse,prop:psi-inverse,thm:phi-maj-inv}, we prove that \(\phi\) is a bijection that preserves \(\content\), \(\inv\), and \(\maj\).
Let's highlight the parts that are straightforward from the definition via blocks.

We verify that both \(\phi\) and \(\psi\) are bijections by noticing that \(\phi(\phi(T)) = T\) and \(\psi(\psi(T)) = T\).
The local restrictions guarantee that \(\content(T) = \content(U)\).

Moreover, it is straightforward to verify that the number of inversion triples within columns \(i\) and \(i+1\) is the same in \(T\) and \(\phi(T)\), since the number of inversion triples within each block is preserved, and the choice of swapping or not swapping each block is made precisely to ensure that the number of inversion triples between different blocks is also preserved.
It is not hard to see that this implies that \(\inv(T) = \inv(\phi(T))\).

The fact that \(\maj(T) = \maj(\phi(T))\) is less straightforward,
since we haven't made any direct attempt to control the major index in our definition of \(\phi\).
We prove this in \Cref{thm:phi-maj-inv}.

\subsection{Row-by-row algorithm} \label{subsec:row-by-row}
It turns out that the bijections \(\phi\) and \(\psi\) are the same as the bijections defined in \cite{KLO22}.
We now present the algorithm from \cite{KLO22}, in a clearer form than the original exposition.

Let \(h = \min(\alpha_i, \alpha_{i+1})\).
Let \(T \in \Fill(\alpha)\).
We describe the algorithm to compute \(U = \phi(T)\) or \(U = \psi(T)\).
The algorithm uses a \vocab{coin} with two sides, \(\coinphi\) and \(\coinpsi\), as a bookkeeping device.
Each row from \(1\) to \(h\) is, at some point of the algorithm, assigned a \vocab{coin value} in \(\{\coinphi, \coinpsi\}\).
Initially, assign the coin value of row \(1\) to be \(\coinphi\) if computing \(U = \phi(T)\), or \(\coinpsi\) if computing \(U = \psi(T)\).
We denote by \(\operatorname{coin}_r\) the coin value assigned to row \(r\).

In addition to the coin value, each row from \(1\) to \(h\) is labeled from \(\{\labelN, \labelS, \labelC\}\).
A label of \(\labelN\) means the entries in row \(r\) of columns \(i\) and \(i+1\) in \(T\) and \(U\) are \emph{not} swapped;
a label of \(\labelS\) means they are \emph{swapped};
and a label of \(\labelC\) indicates that the behavior for row \(r\) should \emph{copy} the behavior for the row above it.
We denote by \(\operatorname{label}_r\) the label assigned to row \(r\).

We process the rows iteratively from row \(1\) up to row \(h\).
For each \(r \in \{1,2,\dots, h-1\}\),
we assume that the coin value \(\operatorname{coin}_r\) has already been assigned.
Let \( a = T(i,r+1)\), \(b = T(i+1,r+1)\), \(c = T(i,r)\), and \(d = T(i+1,r)\).
Note that \(\ind{a, d} = \ind{b, c} \neq \ind{a, c} = \ind{b, d}\) is impossible by transitivity,
thus \( \ind{a,d} + \ind{b,c} - \ind{a,c} - \ind{b,d} \in \{1, 0, -1\} \).

\begin{description}
    \item[Case I.] If 
    \(\ind{a,d} + \ind{b,c} - \ind{a,c} - \ind{b,d} = \pm 1\),
    (cases 3, 4, 9, 10 in \cite{KLO22}),  
    then set \(\operatorname{label}_r = \labelC\).
    Moreover, set \(\operatorname{coin}_{r+1} = \operatorname{coin}_r\) if \(\chi_{ab} = \chi_{cd}\) (cases 3, 9 in \cite{KLO22}),
    and set \(\operatorname{coin}_{r+1} \neq \operatorname{coin}_r\) if \(\chi_{ab} \neq \chi_{cd}\) (cases 4, 10 in \cite{KLO22}).

    \item[Case II.] If
    \(\ind{a,c} = \ind{a,d} = \ind{b,c} = \ind{b,d}\)
    (cases 1, 2, 11, 12 in \cite{KLO22}),
    then set \(\operatorname{label}_r = \labelN\) if \(\operatorname{coin}_r = \coinphi\), and set \(\operatorname{label}_r = \labelS\) if \(\operatorname{coin}_r = \coinpsi\).
    Moreover, set \(\operatorname{coin}_{r+1} = \coinphi\).
    
    \item[Case III.] If 
    \(\ind{a,c} = \ind{b,c} \neq \ind{a,d} = \ind{b,d}\)  
    (cases 7, 8 in \cite{KLO22}),
    then set \(\operatorname{label}_r = \labelS\) if \(\operatorname{coin}_r = \coinphi\), and set \(\operatorname{label}_r = \labelN\) if \(\operatorname{coin}_r = \coinpsi\).
    Moreover, set \(\operatorname{coin}_{r+1} = \coinphi\).
    
    \item[Case IV.] If
    \(\ind{a,c} = \ind{a,d} \neq \ind{b,c} = \ind{b,d}\)
    (cases 5, 6 in \cite{KLO22}),
    then set \(\operatorname{label}_r = \labelN\) if \(\operatorname{coin}_r = \coinphi\), and set \(\operatorname{label}_r = \labelS\) if \(\operatorname{coin}_r = \coinpsi\).
    Moreover, set \(\operatorname{coin}_{r+1} = \coinpsi\).
\end{description}

Finally, we process row \(r = h\), assuming that the coin value \(\operatorname{coin}_h\) has already been assigned.
Let \(a = T(i,h+1)\) if \(h = \alpha_{i+1} < \alpha_i\) or \(a = T(i+1,h+1)\) if \(h = \alpha_{i} < \alpha_{i+1}\), i.e., \(a\) is the only entry in row \(h+1\) and in one of the two columns \(i\) or \(i+1\).
Let \(c = T(i,h)\), and let \(d = T(i+1,h)\).

\begin{description}
    \item[Base I.] If \(\ind{a,c} \neq \ind{a,d}\),
    then set \(\operatorname{label}_r = \labelS\) if \(\operatorname{coin}_r = \coinphi\), and set \(\operatorname{label}_r = \labelN\) if \(\operatorname{coin}_r = \coinpsi\).

    \item[Base II.] If \(\ind{a,c} = \ind{a,d}\), then set \(\operatorname{label}_r = \labelN\) if \(\operatorname{coin}_r = \coinphi\), and set \(\operatorname{label}_r = \labelS\) if \(\operatorname{coin}_r = \coinpsi\).
\end{description}

\begin{definition}[The bijections \(\phi\) and \(\psi\) via the row-by-row algorithm] \label{def:phi-psi-via-row-by-row}
    Let \(T \in \Fill(\alpha)\).
    The row-by-row algorithm described above, with initial coin value \(\coinphi\) (resp.~\(\coinpsi\)), produces a label for each row \(r \in \interval{h}\).
    We define \(U = \phi(T) \in \Fill(s_i \permact \alpha)\) (resp.~\(U = \psi(T) \in \Fill(s_i \permact \alpha)\))
    by requiring the local restrictions (i.e., \eqref{eq:restriction-other-columns} and \eqref{eq:restriction-swap-per-row}) and requiring that, for each row \(r \in \interval{h}\),
    \begin{itemize}

\end{definition}

The row-by-row algorithm may be described recursively instead of iteratively.

Let \(T \in \Fill(\alpha)\) with \(h = \min(\alpha_i, \alpha_{i+1})\).
Assign the coin value of row \(1\) to be \(\coinphi\) if computing \(U = \phi(T)\), or \(\coinpsi\) if computing \(U = \psi(T)\).
If \(h=1\), then assign the label of this row according to Base~I or~II.
Otherwise, assign the label of row \(1\) and the coin value of row \(2\) according one of Cases~I--IV.
Remove the first row to obtain the sub-filling \(T'\), and apply the algorithm recursively to \(T'\) with this new coin value, obtaining the labels for rows \(2,\dots, h\).
Once all labels are determined, construct \(U\) in a final pass by applying the rules for \(\labelN,\labelS,\labelC\) as before.

It is straightforward to verify that both the iterative and recursive descriptions of the algorithm are equivalent.
This recursive interpretation is particularly useful for proving properties about the bijections \(\phi\) and \(\psi\), as it allows us to use induction on the number of rows in the filling.

\subsection{Equivalence of the two definitions} \label{subsec:equivalence-of-definitions}

We now show that the two definitions of \(\phi\) and \(\psi\) given in \Cref{subsec:block-structure,subsec:row-by-row} are equivalent.
We do so by applying the row-by-row algorithm and observing what happens to each block.

\begin{proposition} \label{prop:equivalence-of-definitions}
    The definitions of \(\phi\) and \(\psi\) via blocks (\Cref{def:phi-via-blocks,def:psi-via-blocks}) are equivalent to the definitions via the row-by-row algorithm (\Cref{def:phi-psi-via-row-by-row}).
\end{proposition}

\begin{proof}
    Let \(\phi\) be defined as in \Cref{def:phi-psi-via-row-by-row}.
    We show that \(\phi\) satisfies the conditions in \Cref{def:phi-via-blocks}.
    Let \(T \in \Fill(\alpha)\).
    Consider a block \([r_1, r_2]\) of \(T\) with signature entries \((a, b, c, d, e, f)\).

    First, if \(r_1 > 1\),
    consider the row-by-row algorithm at row \(r_1 - 1\), and let's compute the coin value for row \(r_1\).
    Since \(r_1 - 1\) and \(r_1\) are not intertwined,
    we are in Case~II, III, or~IV.
    Therefore, we have
    \begin{equation} \label{eq:coin-r1}
        \operatorname{coin}_{r_1} = \coinphi
        \quad \text{if and only if} \quad
        \ind{c, f} - \ind{d, f} = 0.
    \end{equation}
    If \(r_1 = 1\), then by definition, \(\operatorname{coin}_{r_1} = \coinphi\),
    and by convention, \(\ind{c, f} - \ind{d, f} = 0\) since \(f = \infty\).
    Thus, \eqref{eq:coin-r1} holds independently of whether \(r_1 = 1\) or \(r_1 > 1\).

    Second, for each \(r \in \{r_1, r_1 + 1, \dots, r_2 - 1\}\),
    since rows \(r\) and \(r + 1\) are intertwined,
    the row-by-row algorithm falls into Case~I, \(\operatorname{label}_r = \labelC\).
    Thus, the rows in the block all copy the behavior of row \(r_2\) and the label of row \(r_2\) determines whether the block is swapped or not swapped,
    that is, the entire block is swapped if and only if \(\operatorname{label}_{r_2} = \labelS\).
    
    Moreover, \(\operatorname{coin}_{r+1} = \operatorname{coin}_r\) if and only if \(\ind{{T(i,r+1)},{T(i+1,r+1)}} = \ind{{T(i,r)},{T(i+1,r)}}\).
    Stringing these together, we have
    \begin{equation} \label{eq:coin-r2-coin-r1}
        \operatorname{coin}_{r_2} = \operatorname{coin}_{r_1}
        \quad \text{if and only if} \quad
        \ind{a, b} - \ind{c, d} = 0.
    \end{equation}

    Third, consider the row-by-row algorithm at row \(r_2\), and let's compute the label for row \(r_2\).
    If \(r_2 = h\), then we are in Base~I or~II,
    otherwise, rows \(r_2\) and \(r_2 + 1\) are not intertwined,
    and thus we are in Case~II, III, or~IV.
    In all of these cases, we have
    \begin{equation} \label{eq:label-r2}
        \operatorname{label}_{r_2} = \labelS
        \quad \text{if and only if} \quad
        (\ind{e, a} - \ind{e, b} = 0) \mathbin{\mathrm{xor}} (\operatorname{coin}_{r_2} = \coinpsi).
    \end{equation}

    Finally, using that \(\operatorname{coin}_{r_2} = \coinpsi\) if and only if \( (\operatorname{coin}_{r_2} = \operatorname{coin}_{r_1} ) \mathbin{\mathrm{xor}} (\operatorname{coin}_{r_1} = \coinphi) \),
    we have
    \begin{align*}
        \operatorname{label}_{r_2} = \labelS
        &\iff (\ind{e, a} - \ind{e, b} = 0) \mathbin{\mathrm{xor}} (\operatorname{coin}_{r_2} = \operatorname{coin}_{r_1} ) \mathbin{\mathrm{xor}} (\operatorname{coin}_{r_1} = \coinphi) \\
        &\iff (\ind{e, a} - \ind{e, b} = 0) \mathbin{\mathrm{xor}} (\ind{a, b} - \ind{c, d} = 0) \mathbin{\mathrm{xor}} (\ind{c, f} - \ind{d, f} = 0) \\
        &\iff \ind{e, a} - \ind{e, b} + \ind{a, b} - \ind{c, d} + \ind{c, f} - \ind{d, f} \equiv 1 \pmod{2} \\
        &\iff \ind{e, a, b} \neq \ind{c, d, f}.
    \end{align*}
    which is the same swapping condition as in \Cref{def:phi-via-blocks}.
    Therefore, the two definitions of \(\phi\) are equivalent.

    The analysis for \(\psi\) is analogous.
    For \(\psi\), \eqref{eq:coin-r1} becomes
    \begin{equation*}
        \operatorname{coin}_{r_1} = \coinphi
        \quad \text{if and only if} \quad
        (\ind{c, f_1} - \ind{d, f_1} = 0) \mathrel{\text{xor}} (r_1 = 1),
    \end{equation*}
    while \eqref{eq:coin-r2-coin-r1} and \eqref{eq:label-r2} remain unchanged.
    Therefore, the conclusion becomes
    \begin{equation*}
        \operatorname{label}_{r_2} = \labelS
        \iff (\ind{e_1, a, b} \neq \ind{c, d, f_1}) \mathbin{\mathrm{xor}} (r_1 = 1),
    \end{equation*}
    which is the same swapping condition as in \Cref{def:psi-via-blocks}.
    Therefore, the two definitions of \(\psi\) are equivalent.
\end{proof}

\subsection{Properties of \texorpdfstring{\(\phi\)}{φ} and \texorpdfstring{\(\psi\)}{ψ} via the block structure algorithm} \label{subsec:properties-of-bijections}

As an abuse of notation, let \(\phi\) denote both the function \(\phi \colon \Fill(\alpha) \to \Fill(s_i \permact \alpha)\) and the function \(\phi \colon \Fill(s_i \permact \alpha) \to \Fill(\alpha)\).
The intended use of \(\phi\) is always clear from the shape of the input filling.

\begin{proposition} \label{prop:phi-inverse}
    The maps \(\phi \colon \Fill(\alpha) \to \Fill(s_i \permact \alpha)\) and \(\phi \colon \Fill(s_i \permact \alpha) \to \Fill(\alpha)\) are inverses of each other.
    Moreover, \(\phi\) is a bijection.
\end{proposition}

\begin{proof}[Proof]
    Let \(T \in \Fill(\alpha)\).
    Let \(U = \phi(T)\) and \(T' = \phi(U)\).
    Since the condition of two rows being intertwined only depends on the set of entries in two rows, not the order of the entries,
    the block structure of \(T\), \(U\), and \(T'\) are the same.
    Let \([r_1, r_2]\) be a block of \(T \in \Fill(\alpha)\) with signature entries \((a, b, c, d, e, f)\).

    If \(a = b\) or \(c = d\), then the block consists of a single row, since a row of equal entries cannot be intertwined with any other row.
    Therefore, the columns \(i\) and \(i+1\) of this row are identical in \(T\), \(U\), and \(T'\).
    Thus, we may assume that \(a \neq b\) and \(c \neq d\).

    By~\Cref{def:phi-via-blocks}, the rows in \([r_1, r_2]\) are swapped between \(T\) and \(U\) if and only if \( \ind{c, d, f} \neq \ind{e, a, b} \).

    If this condition is false, then the rows in \([r_1, r_2]\) are not swapped between \(T\) and \(U\).
    Therefore, \([r_1, r_2]\) is a block of \(U\) with signature entries \((a, b, c, d, e, f)\), regardless of whether the blocks immediately above or below are swapped or not.
    Now, again by~\Cref{def:phi-via-blocks}, the rows in \([r_1, r_2]\) are swapped between \(U\) and \(T'\) if and only if \(\ind{c, d, f} \neq \ind{e, a, b}\), which is false.
    Therefore, the rows in \([r_1, r_2]\) are not swapped between \(U\) and \(T'\), and consequently, \(T\) and \(T'\) are identical in this block.

    If this condition is true, then the rows in \([r_1, r_2]\) are swapped between \(T\) and \(U\).
    Therefore, \([r_1, r_2]\) is a block of \(U\) with signature entries \((b, a, d, c, e, f)\), regardless of whether the blocks immediately above or below are swapped or not.
    Now, by~\Cref{def:phi-via-blocks}, the rows in \([r_1, r_2]\) are swapped between \(U\) and \(T'\) if and only if \(\ind{d, c, f} \neq \ind{e, b, a}\).
    Because \(a \neq b\) and \(c \neq d\), we have \(\ind{d, c, f} = 1 - \ind{c, d, f}\) and \(\ind{e, b, a} = 1 - \ind{e, a, b}\).
    Therefore, \(\ind{d, c, f} \neq \ind{e, b, a}\) is equivalent to \(\ind{c, d, f} \neq \ind{e, a, b}\), which is true.
    Thus, the rows in \([r_1, r_2]\) are swapped between \(U\) and \(T'\).
    Since the rows in \([r_1, r_2]\) are also swapped between \(T\) and \(U\), then \(T\) and \(T'\) are identical in this block.

    Therefore, for each block of \(T\), the corresponding block of \(T'\) is identical.
    All other entries of \(T\) and \(T'\) outside of blocks are equal by the local restrictions.
    Thus, \(T = T'\), and consequently \(\phi \colon \Fill(\alpha) \to \Fill(s_i \permact \alpha)\) is the inverse of \(\phi \colon \Fill(s_i \permact \alpha) \to \Fill(\alpha)\).
    Since \(\phi\) has an inverse, it is a bijection.
\end{proof}

\begin{proposition} \label{prop:psi-inverse}
    The maps \(\psi \colon \Fill(\alpha) \to \Fill(s_i \permact \alpha)\) and \(\psi \colon \Fill(s_i \permact \alpha) \to \Fill(\alpha)\) are inverses of each other.
    Moreover, \(\psi\) is a bijection.
\end{proposition}

The proof of \Cref{prop:psi-inverse} is analogous to the proof of \Cref{prop:phi-inverse}, with a minor modification to account for the different swapping condition of the bottommost block.

In \Cref{thm:phi-maj-inv,thm:psi-maj-inv}, we describe how the bijections \(\phi\) and \(\psi\) interact with the statistics \(\inv\) and \(\maj\).
Before then, we need \Cref{lem:leg-cancelation,lem:arm-cancelation} that describe certain cancelation properties of the \(\leg\) and \(\arm\) statistics.

Given a box \(u = (j, r)\), let \(s_i \cdot u = (s_i(j), r)\) be the box obtained by applying the transposition \(s_i\) to the column index of \(u\).

\begin{lemma} \label{lem:leg-cancelation}
    Let \(\alpha\) be a composition.
    Let \(u, v \in \dg(\alpha)\) be boxes in columns \(i\) and \(i+1\), respectively, such that \(u, v \in \dg(s_i \permact \alpha)\).
    Then,
    \begin{equation*}
        \leg_{\alpha}(u) + \leg_{\alpha}(v)
        =
        \leg_{s_i \permact \alpha}(u) + \leg_{s_i \permact \alpha}(v).
    \end{equation*}
\end{lemma}
\begin{proof}
    Let \(u = (i, r)\) and \(v = (i+1, r')\).
    Recall that \( \leg_{\alpha}(i, r) = \alpha_i - r \).
    Therefore, both quantities are equal to \( \alpha_i + \alpha_{i+1} - r - r'. \)
\end{proof}

\begin{lemma} \label{lem:arm-cancelation}
    Let \(\alpha\) be a composition.
    Let \(u, v \in \dg(\alpha)\) be boxes in columns \(i\) and \(i+1\), respectively, such that \(u, v \in \dg(s_i \permact \alpha)\).
    Then,
    \begin{equation*}
        \arm_{\alpha}(u) + \arm_{\alpha}(v)
        =
        \arm_{s_i \permact \alpha}(u) + \arm_{s_i \permact \alpha}(v).
    \end{equation*}
\end{lemma}
\begin{proof}
    Let \(u = (i, r)\) and \(v = (i+1, r')\).
    Recall that 
    \begin{equation*}
        \arm_{\alpha}(i, r) = |\{j < i : \alpha_j \in \interval{r-1, \alpha_i - 1}\}| + |\{j > i : \alpha_j \in \interval{r, \alpha_i}\}|.
    \end{equation*}
    Let \(\chi_S(x)\) be the indicator function that is \(1\) if \(x \in S\) and \(0\) otherwise.
    Therefore, for all both quantities, the contribution from \(j < i\) is
    \begin{equation*}
        \chi_{\interval{r-1, \alpha_i - 1}}(\alpha_j) + \chi_{\interval{r', \alpha_{i+1} - 1}}(\alpha_j) =
        \chi_{\interval{r-1, \alpha_{i+1} - 1}}(\alpha_j) + \chi_{\interval{r', \alpha_i - 1}}(\alpha_j),
    \end{equation*}
    the contribution from \(j > i+1\) is
    \begin{equation*}
        \chi_{\interval{r, \alpha_i}}(\alpha_j) + \chi_{\interval{r' - 1, \alpha_{i+1}}}(\alpha_j) =
        \chi_{\interval{r', \alpha_i}}(\alpha_j) + \chi_{\interval{r - 1, \alpha_{i+1}}}(\alpha_j),
    \end{equation*}
    and the combined contribution from \(j = i\) and \(j = i+1\) is one.
    Hence, all three quantities are equal.
\end{proof}

\begin{theorem}[{\cite[Proposition~4.3]{KLO22}}] \label{thm:phi-maj-inv}
    The bijection \(\phi\) preserves the statistics \(\maj\) and \(\inv\); that is, for all \(T \in \Fill(\alpha)\), if \(U = \phi(T)\), then
    \begin{equation} \label{eq:phi-maj}
        \maj(T) = \maj(U)
    \end{equation}
    and
    \begin{equation} \label{eq:phi-inv}
        \inv(T) = \inv(U).
    \end{equation}
\end{theorem}

\begin{proof}[Proof of \eqref{eq:phi-maj}, independent of \textup{\cite{KLO22}}]
    Without loss of generality, assume that \(\alpha_i > \alpha_{i+1}\).
    Let \(T \in \Fill(\alpha)\) and \(U = \phi(T)\).
    We first study how \(\Des(T)\) and \(\Des(U)\) relate to each other.

    Clearly, for all boxes \(u\) not in columns \(i\) or \(i+1\), we have \(u \in \Des(T)\) if and only if \(u \in \Des(U)\), by \eqref{eq:restriction-other-columns}.
    Since \(\leg_\alpha(u) = \leg_{s_i \permact \alpha}(u)\) for such boxes \(u\),
    these descents contribute equally to \(\maj(T)\) and \(\maj(U)\).
    Moreover, for all boxes \(u = (i, r)\) with \(r \geq \alpha_{i+1} + 2\),
    we have \(u \in \Des(T)\) if and only if \(s_i \cdot u \in \Des(U)\), by \eqref{eq:restriction-swap-per-row}.
    Note that \(\leg_\alpha(u) = \alpha_{i} - r = \leg_{s_i \permact \alpha}(s_i \cdot u)\) for such boxes \(u\),
    so these descents contribute equally to \(\maj(T)\) and \(\maj(U)\).
    Therefore, it suffices to show that the contribution to \(\maj\) from the boxes in columns \(i\) and \(i+1\) at rows \(2\) through \(\alpha_{i+1} + 1\) are the same in both \(T\) and \(U\).
    
    We say that a row \(r \in \interval{2, \alpha_{i+1}}\) is \vocab{balanced} in \(T\) if both or neither of the boxes \((i, r)\) and \((i+1, r)\) are descents of \(T\);
    is \vocab{left-unbalanced} in \(T\) if \((i, r)\) is a descent of \(T\) but \((i+1, r)\) is not;
    is \vocab{right-unbalanced} in \(T\) if \((i+1, r)\) is a descent of \(T\) but \((i, r)\) is not.

    Let \(r \in \interval{2, \alpha_{i+1}}\) be a balanced row in \(T\).
    Let \(a = T(i, r)\), \(b = T(i+1, r)\), \(c = T(i, r-1)\), and \(d = T(i+1, r-1)\).
    The number of descents at row \(r\) in \(T\) is \(\ind{a, c} + \ind{b, d} \in \{0, 2\}\).
    On the one hand, if rows \(r\) and \(r-1\) have the same swapping behavior in \(T\) and \(U\), then number of descents at row \(r\) in \(U\) is also \(\ind{a, c} + \ind{b, d}\), the same as in \(T\).
    On the other hand, if rows \(r\) and \(r-1\) have different swapping behavior between \(T\) and \(U\), 
    then \(r\) and \(r-1\) are not intertwined in \(T\),
    so \(\ind{a, c} + \ind{b, d} = \ind{a, d} + \ind{b, c}\).
    In this case,
    the number of descents at row \(r\) in \(U\) is given by \(\ind{a, d} + \ind{b, c}\), which is equal to \(\ind{a, c} + \ind{b, d}\), the same as in \(T\).
    Therefore, for any balanced row \(r\), the number of descents at row \(r\) is the same in \(T\) and \(U\) (either \(0\) or \(2\)).
    Moreover, by \Cref{lem:leg-cancelation}, these descents contribute equally to \(\maj(T)\) and \(\maj(U)\).

    Therefore, it suffices to show that the contribution to \(\maj\) from unbalanced rows in \(T\) is the same as the contribution from unbalanced rows in \(U\).
    Each of the unbalanced rows in \([2, \alpha_{i+1} + 1]\) will be \vocab{assigned} to a unique block of \(T\), which is not necessarily the block that contains that row.
    We describe how the assignment is done.

    As an edge case, row \(\alpha_{i+1} + 1\) is assigned to the topmost block.
    Let \(r \in \interval{2, \alpha_{i+1}}\).
    Let \(a = T(i, r)\), \(b = T(i+1, r)\), \(c = T(i, r-1)\), and \(d = T(i+1, r-1)\).
    
    First, assume rows \(r-1\) and \(r\) are not intertwined in \(T\), that is \(\ind{a, c} + \ind{b, d} = \ind{a, d} + \ind{b, c}\).
    Note that row \(r\) is balanced in \(T\) if and only if \(\ind{a, c} = \ind{a, d} = \ind{b, d} = \ind{b, c}\).
    Assume row \(r\) is unbalanced in \(T\), so that \(\ind{a, c} + \ind{b, d} = 1\).
    Hence, we also have \(\ind{a, d} + \ind{b, c} = 1\), so independently of whether each of the blocks is swapped or not, exactly one of \((i, r)\) and \((i + 1, r)\) is a descent of \(U\) as well.
    If \(\ind{a, c} = \ind{a, d} \neq \ind{b, d} = \ind{b, c}\),
    then the unique descent in \(T\) and the unique descent in \(U\) are in the column that contains the largest of \(a\) and \(b\).
    In other words, \(r\) is unbalanced on the same side in both \(T\) and \(U\) if \(r\) is not swapped between \(T\) and \(U\),
    and \(r\) is unbalanced on opposite sides in \(T\) and \(U\) if \(r\) is swapped between \(T\) and \(U\).
    In this case, we \vocab{assign} row \(r\) to the block \(B_1\), which is the block that determines the swapping behavior of row \(r\).
    Otherwise, if \(\ind{a, c} = \ind{b, c} \neq \ind{a, d} = \ind{b, d}\),
    then the unique descent in \(T\) and the unique descent in \(U\) are in the column that contains the smallest of \(c\) and \(d\).
    In other words, \(r\) is unbalanced on the same side in both \(T\) and \(U\) if \(r-1\) is not swapped between \(T\) and \(U\),
    and \(r\) is unbalanced on opposite sides in \(T\) and \(U\) if \(r-1\) is swapped between \(T\) and \(U\).
    In this case, we \vocab{assign} row \(r\) to the block \(B_2\), which is the block that determines the swapping behavior of row \(r-1\).

    Second, assume rows \(r-1\) and \(r\) are intertwined in \(T\), meaning that both rows belong to the same block \(B\), and that \(\ind{a, c} + \ind{b, d} + \ind{a, d} + \ind{b, c}\) is odd.
    we have \(\ind{a, c} = \ind{b, d}\).
    Using the fact that \(\ind{a, c} + \ind{b, d} + \ind{a, d} + \ind{b, c}\) is odd,
    we can conclude that \(\ind{a, b} = \ind{c, d}\) in this case, that is, \(\ind{r} = \ind{r-1}\).
    If row \(r\) is left-unbalanced with respect to \(T\), meaning that only \((i, r)\) is a descent of \(T\) (equivalently \(\ind{a, c} = 1\) and \(\ind{b, d} = 0\)), then we have
    \(\ind{r} = \ind{a, b} = 1\) and \(\ind{r-1} = \ind{c, d} = 0\).
    If row \(r\) is right-unbalanced with respect to \(T\), meaning that only \((i+1, r)\) is a descent of \(T\) (equivalently \(\ind{a, c} = 0\) and \(\ind{b, d} = 1\)), then we have
    \(\ind{r} = \ind{a, b} = 0\) and \(\ind{r-1} = \ind{c, d} = 1\).
    Note that we can determine whether such row \(r\) is balanced, left-unbalanced, or right-unbalanced by looking only at \(\ind{r-1}\) and \(\ind{r}\): if \(\ind{r-1} = \ind{r}\), then row \(r\) is balanced; if \(\ind{r-1} = 0\) and \(\ind{r} = 1\), then row \(r\) is left-unbalanced; and if \(\ind{r-1} = 1\) and \(\ind{r} = 0\), then row \(r\) is right-unbalanced.

    Let \(\Des(B; T)\) be the set of descents of \(T\) in rows assigned to the block \(B\).
    We define the \vocab{major contributions} of a block \(B\) with respect to the fillings \(T\) and \(U\) as
    \begin{align*}
        \maj(B; T) &= \sum_{u \in \Des(B; T)} (\leg_{\alpha}(u) + 1), \\
        \maj(B; U) &= \sum_{u \in \Des(B; U)} (\leg_{s_i \permact \alpha}(u) + 1).
    \end{align*}
    We claim that for every block \(B\), we have \(\maj(B; T) = \maj(B; U)\).
    Let \(B = [r_1, r_2]\) be a block of \(T\).

    Assume that \(B\) is swapped between \(T\) and \(U\).
    Then, if \(u\) is a descent assigned to \(B\) in \(T\), then \(s_i \cdot u\) is a descent assigned to \(B\) in \(U\), and vice versa,
    regardless of whether the descent is in row \(r_1\), row \(r_2 + 1\), or any row in between.
    Since \(\leg_{\alpha}(u) = \leg_{s_i \permact \alpha}(s_i \cdot u)\) for all boxes \(u\) in columns \(i\) or \(i+1\),
    we have \(\maj(B; T) = \maj(B; U)\) for all blocks \(B\) that are swapped between \(T\) and \(U\).

    Now, assume that \(B = [r_1, r_2]\) is not swapped between \(T\) and \(U\), and that it is not the topmost block.
    Let \((a, b, c, d, e, f)\) be signature entries of \(B\).
    This means that \(\ind{c, d, f} = \ind{e, a, b}\), that is,
    \begin{equation*}
        \ind{c, d} + \ind{d, f} - \ind{c, f} = \ind{e, a} + \ind{a, b} - \ind{e, b}.
    \end{equation*}
    We know that unbalanced rows \(r_1 + 1\) through \(r_2\) are assigned to \(B\).
    Let's apply the assigning rules to rows \(r_1\) and \(r_2 + 1\).
    The only possibility for row \(r_1\) to be unbalanced and assigned to \(B\) is when \(\ind{c, f} \neq \ind{d, f}\), in which case row \(r_1\) is unbalanced to the side that contains the largest of \(c\) and \(d\).
    Similarly, the only possibility for row \(r_2 + 1\) to be unbalanced and assigned to \(B\) is when \(\ind{e, a} \neq \ind{e, b}\), in which case row \(r_2 + 1\) is unbalanced to the side that contains the smallest of \(a\) and \(b\).

    First, assume \(\ind{a, b} = \ind{c, d}\), that is, \(\ind{r_2} = \ind{r_1}\).
    In the sequence \((\ind{r_1}, \ind{r_1+1}, \dots, \ind{r_2})\), each time the indicator does not change, the corresponding row is balanced, and each time the indicator changes from \(0\) to \(1\) (resp. from \(1\) to \(0\)), the corresponding row is left-unbalanced (resp. right-unbalanced).
    Since \(\ind{r_2} = \ind{r_1}\), the number of left-unbalanced rows is equal to the number of right-unbalanced rows among rows \(r_1 + 1\) through \(r_2\).
    Moreover, we have \(\ind{d, f} - \ind{c, f} = \ind{e, a} - \ind{e, b}\).
    This means that \(\ind{d, f} = \ind{c, f}\) if, and only if, \(\ind{e, a} = \ind{e, b}\).
    If both are true, then rows \(r_1\) and \(r_2 + 1\) are not assigned to \(B\), and consequently, the numbers of left-unbalanced and right-unbalanced rows assigned to \(B\) are equal.
    If both are false, then rows \(r_1\) and \(r_2 + 1\) are unbalanced and assigned to \(B\), with row \(r_1\) unbalanced to the side of the largest of \(c\) and \(d\), and row \(r_2 + 1\) unbalanced to the side of the smallest of \(a\) and \(b\).
    Since \(\ind{a, b} = \ind{c, d}\), the largest of \(c\) and \(d\) is in the opposite column as the smallest of \(a\) and \(b\),
    and consequently, row \(r_1\) is left-unbalanced while row \(r_2 + 1\) is right-unbalanced, or vice versa.
    Therefore, in this case, the number of left-unbalanced rows assigned to \(B\) is equal to the number of right-unbalanced rows assigned to \(B\) as well.

    Second, assume \(\ind{a, b} \neq \ind{c, d}\), that is, \(\ind{r_2} \neq \ind{r_1}\).
    Moreover, assume \(\ind{a, b} = 1\) and \(\ind{c, d} = 0\).
    In the sequence \((\ind{r_1}, \ind{r_1+1}, \dots, \ind{r_2})\), each time the indicator does not change, the corresponding row is balanced, and each time the indicator changes from \(0\) to \(1\) (resp. from \(1\) to \(0\)), the corresponding row is left-unbalanced (resp. right-unbalanced).
    Since this sequence starts with \(0\) and ends with \(1\), there is exactly one more left-unbalanced row than right-unbalanced rows among rows \(r_1 + 1\) through \(r_2\).
    Moreover, we have \(\ind{d, f} - \ind{c, f} \neq \ind{e, a} - \ind{e, b}\).
    This means that exactly one of \(\ind{d, f} \neq \ind{c, f}\) and \(\ind{e, a} \neq \ind{e, b}\) is true.
    If \(\ind{d, f} \neq \ind{c, f}\) and \(\ind{e, a} = \ind{e, b}\), then row \(r_1\) is unbalanced to the side of the largest of \(c\) and \(d\), which is \(d\) since \(\ind{c, d} = 0\), that is, row \(r_1\) is right-unbalanced; while row \(r_2 + 1\) is not assigned to \(B\).
    Therefore, with this extra right-unbalanced row assigned to \(B\),
    the numbers of left-unbalanced and right-unbalanced rows assigned to \(B\) are equal.
    If \(\ind{d, f} = \ind{c, f}\) and \(\ind{e, a} \neq \ind{e, b}\), then row \(r_2 + 1\) is unbalanced to the side of the smallest of \(a\) and \(b\), which is \(a\) since \(\ind{a, b} = 1\), that is, row \(r_2 + 1\) is left-unbalanced; while row \(r_1\) is not assigned to \(B\).
    Therefore, with this extra left-unbalanced row assigned to \(B\),
    the numbers of left-unbalanced and right-unbalanced rows assigned to \(B\) are equal as well.
    The argument for the case \(\ind{a, b} = 0\) and \(\ind{c, d} = 1\) is analogous, by swapping the roles of left and right throughout.

    Therefore, in all cases, the number of left-unbalanced rows assigned to \(B\) is equal to the number of right-unbalanced rows assigned to \(B\).
    This means that we can pair each left-unbalanced row assigned to \(B\) with a unique right-unbalanced row assigned to \(B\).
    By \Cref{lem:leg-cancelation},
    the contributions to \(\maj(B; T)\) and \(\maj(B; U)\) from each such pair are equal, and consequently, \(\maj(B; T) = \maj(B; U)\) for all non-topmost blocks \(B\) that are not swapped between \(T\) and \(U\).

    Finally, assume that \(B = [r_1, \alpha_{i+1}]\) is the topmost block of \(T\), and that it is not swapped between \(T\) and \(U\).
    Let \((a, b, c, d, e, f)\) be signature entries of \(B\).
    This means that \(\ind{c, d} + \ind{d, f} - \ind{c, f} = \ind{e, a} + \ind{a, b} - \ind{e, b}\).
    We know that unbalanced rows \(r_1 + 1\) through \(r_2\) are assigned to \(B\).
    We also know that row \(\alpha_{i+1} + 1\) is assigned to \(B\), as an edge case.
    The box \((i, \alpha_{i+1} + 1)\) is a descent of \(T\) if and only if \(\ind{e, a} = 1\), in which case its contribution to \(\maj(T)\) is \(\alpha_i - \alpha_{i+1}\).
    Similarly, the box \((i+1, \alpha_{i+1} + 1)\) is a descent of \(U\) if and only if \(\ind{e, b} = 1\), in which case its contribution to \(\maj(U)\) is \(\alpha_i - \alpha_{i+1}\).
    Finally, just as in the non-topmost case, the only possibility for row \(r_1\) to be unbalanced and assigned to \(B\) is when \(\ind{c, f} \neq \ind{d, f}\), in which case row \(r_1\) is unbalanced to the side that contains the largest of \(c\) and \(d\).

    First, assume \(\ind{a, b} = \ind{c, d}\), that is, \(\ind{r_2} = \ind{r_1}\).
    As in the non-topmost case, the number of left-unbalanced rows assigned to \(B\) is equal to the number of right-unbalanced rows assigned to \(B\) among rows \(r_1 + 1\) through \(\alpha_{i+1}\).
    As before, these can be paired up so that the contribution to \(\maj(B; T)\) and \(\maj(B; U)\) from each such pair are equal.
    The remaining contributions to \(\maj(B; T)\) and \(\maj(B; U)\) come from rows \(r_1\) and \(\alpha_{i+1} + 1\).
    We have \(\ind{d, f} - \ind{c, f} = \ind{e, a} - \ind{e, b}\), which means that \(\ind{d, f} = \ind{c, f}\) if, and only if, \(\ind{e, a} = \ind{e, b}\).
    If both are true, then \(r_1\) is not assigned to \(B\),
    and the box \((i, \alpha_{i+1} + 1)\) is a descent of \(T\) if and only if the box \((i+1, \alpha_{i+1} + 1)\) is a descent of \(U\), so their contributions to \(\maj(B; T)\) and \(\maj(B; U)\) are equal (both zero or both \(\alpha_i - \alpha_{i+1}\)).
    Assume both are false.
    If \(\ind{a, b} = \ind{c, d} = 0\),
    the we have \(\ind{e, a} = 1\) and \(\ind{e, b} = 0\),
    so the box \((i, \alpha_{i+1} + 1)\) is a descent of \(T\) while the box \((i+1, \alpha_{i+1} + 1)\) is not a descent of \(U\),
    contributing \(\alpha_i - \alpha_{i+1}\) to \(\maj(B; T)\) and \(0\) to \(\maj(B; U)\).
    Moreover, since \(\ind{c, f} \neq \ind{d, f}\), row \(r_1\) is assigned to \(B\) and is unbalanced to the side of the largest of \(c\) and \(d\), which is \(d\) since \(\ind{c, d} = 0\), so row \(r_1\) is right-unbalanced.
    Thus, row \(r_1\) contributes \(\leg_{\alpha}(i+1, r_1) + 1 = \alpha_{i+1} - r_1 + 1\) to \(\maj(B; T)\) and \(\leg_{s_i \permact \alpha}(i+1, r_1) + 1 = \alpha_{i} - r_1 + 1\) to \(\maj(B; U)\).
    Therefore, since
    \begin{equation*}
        (\alpha_i - \alpha_{i+1}) + (\alpha_{i+1} - r_1 + 1)
        =
        (\alpha_{i} - r_1 + 1),
    \end{equation*}
    the contributions to \(\maj(B; T)\) and \(\maj(B; U)\) from rows \(r_1\) and \(\alpha_{i+1} + 1\) are equal as well.
    If \(\ind{a, b} = \ind{c, d} = 1\),
    then we have \(\ind{e, a} = 0\) and \(\ind{e, b} = 1\),
    so the box \((i, \alpha_{i+1} + 1)\) is not a descent of \(T\) while the box \((i+1, \alpha_{i+1} + 1)\) is a descent of \(U\),
    contributing \(0\) to \(\maj(B; T)\) and \(\alpha_i - \alpha_{i+1}\) to \(\maj(B; U)\).
    Moreover, since \(\ind{c, f} \neq \ind{d, f}\), row \(r_1\) is assigned to \(B\) and is unbalanced to the side of the largest of \(c\) and \(d\), which is \(c\) since \(\ind{c, d} = 1\), so row \(r_1\) is left-unbalanced.
    Thus, row \(r_1\) contributes \(\leg_{\alpha}(i, r_1) + 1 = \alpha_{i} - r_1 + 1\) to \(\maj(B; T)\) and \(\leg_{s_i \permact \alpha}(i, r_1) + 1 = \alpha_{i+1} - r_1 + 1\) to \(\maj(B; U)\).
    Therefore, since
    \begin{equation*}
        (\alpha_{i} - r_1 + 1)
        =
        (\alpha_{i+1} - \alpha_{i}) + (\alpha_{i+1} - r_1 + 1),
    \end{equation*}
    the contributions to \(\maj(B; T)\) and \(\maj(B; U)\) from rows \(r_1\) and \(\alpha_{i+1} + 1\) are equal as well.
    Therefore, we have \(\maj(B; T) = \maj(B; U)\) in this case.
    
    Second, assume \(\ind{a, b} = 1\) and \(\ind{c, d} = 0\), that is, \(\ind{r_1} = 0\) and \(\ind{r_2} = 1\).
    There is exactly one more left-unbalanced row than right-unbalanced rows among rows \(r_1 + 1\) through \(\alpha_{i+1}\).
    We have that exactly one of \(\ind{d, f} \neq \ind{c, f}\) and \(\ind{e, a} \neq \ind{e, b}\) is true.
    If \(\ind{d, f} \neq \ind{c, f}\) and \(\ind{e, a} = \ind{e, b}\),
    then row \(r_1\) is unbalanced to the side of the largest of \(c\) and \(d\), which is \(d\) since \(\ind{c, d} = 0\), so row \(r_1\) is right-unbalanced;
    while the box \((i, \alpha_{i+1} + 1)\) is a descent of \(T\) if and only if the box \((i+1, \alpha_{i+1} + 1)\) is a descent of \(U\), so their contributions to \(\maj(B; T)\) and \(\maj(B; U)\) are equal (both zero or both \(\alpha_i - \alpha_{i+1}\)).
    Therefore, with the extra right-unbalanced row \(r_1\) assigned to \(B\),
    the numbers of left-unbalanced and right-unbalanced rows assigned to \(B\) are equal, and consequently \(\maj(B; T) = \maj(B; U)\).
    If \(\ind{d, f} = \ind{c, f}\) and \(\ind{e, a} \neq \ind{e, b}\),
    then row \(r_1\) is not assigned to \(B\);
    since \(\ind{a, b} = 1\), we have \(\ind{e, a} = 0\) and \(\ind{e, b} = 1\),
    so the box \((i, \alpha_{i+1} + 1)\) is not a descent of \(T\) while the box \((i+1, \alpha_{i+1} + 1)\) is a descent of \(U\),
    contributing \(0\) to \(\maj(B; T)\) and \(\alpha_i - \alpha_{i+1}\) to \(\maj(B; U)\).
    We pair each right-unbalanced row assigned to \(B\) with a unique left-unbalanced row assigned to \(B\),
    except for one extra left-unbalanced row.
    This extra left-unbalanced row \(r\) contributes \(\leg_{\alpha}(i, r) + 1 = \alpha_{i} - r + 1\) to \(\maj(B; T)\) and \(\leg_{s_i \permact \alpha}(i, r) + 1 = \alpha_{i+1} - r + 1\) to \(\maj(B; U)\).
    Therefore, since
    \begin{equation*}
        (\alpha_{i} - r + 1)
        =
        (\alpha_{i+1} - \alpha_{i}) + (\alpha_{i+1} - r + 1),
    \end{equation*}
    the contributions to \(\maj(B; T)\) and \(\maj(B; U)\) from the pair of the extra left-unbalanced row and row \(\alpha_{i+1} + 1\) are equal as well.
    Therefore, we have \(\maj(B; T) = \maj(B; U)\) in this case as well.

    Third, assume \(\ind{a, b} = 0\) and \(\ind{c, d} = 1\), that is, \(\ind{r_1} = 1\) and \(\ind{r_2} = 0\).
    There is exactly one more right-unbalanced row than left-unbalanced rows among rows \(r_1 + 1\) through \(\alpha_{i+1}\).
    We have that exactly one of \(\ind{d, f} \neq \ind{c, f}\) and \(\ind{e, a} \neq \ind{e, b}\) is true.
    If \(\ind{d, f} \neq \ind{c, f}\) and \(\ind{e, a} = \ind{e, b}\),
    then row \(r_1\) is unbalanced to the side of the largest of \(c\) and \(d\), which is \(c\) since \(\ind{c, d} = 1\), so row \(r_1\) is left-unbalanced;
    while the box \((i, \alpha_{i+1} + 1)\) is a descent of \(T\) if and only if the box \((i+1, \alpha_{i+1} + 1)\) is a descent of \(U\), so their contributions to \(\maj(B; T)\) and \(\maj(B; U)\) are equal (both zero or both \(\alpha_i - \alpha_{i+1}\)).
    Therefore, with the extra left-unbalanced row \(r_1\) assigned to \(B\),
    the numbers of left-unbalanced and right-unbalanced rows assigned to \(B\) are equal, and consequently \(\maj(B; T) = \maj(B; U)\).
    If \(\ind{d, f} = \ind{c, f}\) and \(\ind{e, a} \neq \ind{e, b}\),
    then row \(r_1\) is not assigned to \(B\);
    since \(\ind{a, b} = 0\), we have \(\ind{e, a} = 1\) and \(\ind{e, b} = 0\),
    so the box \((i, \alpha_{i+1} + 1)\) is a descent of \(T\) while the box \((i+1, \alpha_{i+1} + 1)\) is not a descent of \(U\),
    contributing \(\alpha_i - \alpha_{i+1}\) to \(\maj(B; T)\) and \(0\) to \(\maj(B; U)\).
    We pair each left-unbalanced row assigned to \(B\) with a unique right-unbalanced row assigned to \(B\),
    except for one extra right-unbalanced row.
    This extra right-unbalanced row \(r\) contributes \(\leg_{\alpha}(i+1, r) + 1 = \alpha_{i+1} - r + 1\) to \(\maj(B; T)\) and \(\leg_{s_i \permact \alpha}(i+1, r) + 1 = \alpha_{i} - r + 1\) to \(\maj(B; U)\).
    Therefore, since
    \begin{equation*}
        (\alpha_{i+1} - r + 1) +
        (\alpha_i - \alpha_{i+1})
        =
        (\alpha_{i} - r + 1),
    \end{equation*}
    the contributions to \(\maj(B; T)\) and \(\maj(B; U)\) from the pair of the extra right-unbalanced row and row \(\alpha_{i+1} + 1\) are equal as well.
    Therefore, we have \(\maj(B; T) = \maj(B; U)\) in this case as well.

    Finally, we have shown that for every block \(B\), we have \(\maj(B; T) = \maj(B; U)\).
    Since each unbalanced row is assigned to exactly one block, we conclude that \(\maj(T) = \maj(U)\), completing the proof.
\end{proof}

\begin{proof}[Proof of \eqref{eq:phi-inv}, independent of \textup{\cite{KLO22}}]
    Without loss of generality, assume that \(\alpha_i > \alpha_{i+1}\).
    Let \(T \in \Fill(\alpha)\) and \(U = \phi(T)\).
    Let \(\beta = \composition{0, \dots, 0, \alpha_{i+1}, \alpha_{i+1}, 0, \dots, 0}\).
    Let \(T_1 \in \Fill(\beta)\) and \(U_1 \in \Fill(s_i \permact \beta)\) be the fillings obtained from \(T\) and \(U\) by restricting to columns \(i\) and \(i+1\).

    Using \Cref{prop:inv-via-inversion-pairs},
    we decompose the inversion statistic as
    \begin{equation*}
		\inv(T) - \inv(U)
        =
        \left(
            \left|\InversionPairs(T)\right|
            - \left|\InversionPairs(U)\right|
        \right)
        -
        \left(
            \sum_{u \in \Des(T)} \arm_{\alpha}(u)
            - \sum_{u \in \Des(U)} \arm_{s_i \permact \alpha}(u)
        \right).
    \end{equation*}

    Note that a pair of boxes both not in columns \(i\) or \(i+1\) is an inversion pair in \(T\) if and only if it is an inversion pair in \(U\).
    Note that a box \((i', r')\) with \(i' \notin \{i, i+1\}\) attacks (resp.~is attacked by) either none or both of \((i, r)\) and \((i+1, r)\).
    Moreover, by \eqref{eq:restriction-swap-per-row}, 
    for each row \(r\), we have
    \begin{equation*}
        \{T(i, r), T(i+1, r)\} = \{U(i, r), U(i+1, r)\}.
    \end{equation*}
    Therefore, the number of inversion pairs in \(T\) involving a box not in columns \(i\) or \(i+1\) and a box in column \(i\) or \(i+1\) is equal to the number of inversion pairs in \(U\) involving a box not in columns \(i\) or \(i+1\) and a box in column \(i\) or \(i+1\).
    Consequently, the difference \(\left|\InversionPairs(T)\right| - \left|\InversionPairs(U)\right| = \left|\InversionPairs(T_1)\right| - \left|\InversionPairs(U_1)\right|\).

    Next, we analyze the difference
    \begin{equation*}
        \sum_{u \in \Des(T)} \arm_{\alpha}(u)
        - \sum_{u \in \Des(U)} \arm_{s_i \permact \alpha}(u).
    \end{equation*}
    Note that a box \(u\) not in columns \(i\) or \(i+1\) has the same arm length in \(\alpha\) and in \(s_i \permact \alpha\).
    Moreover, by \eqref{eq:restriction-other-columns}, such a box \(u\) is a descent of \(T\) if and only if it is a descent of \(U\).
    Therefore, the contributions to the above difference from boxes not in columns \(i\) or \(i+1\) cancel out.
    Consequently, changing the summations to only consider descents in \(\dg(\beta)\) and \(\dg(s_i \cdot \beta)\), respectively, does not change the value of the difference.
    Define
    \begin{equation}
        \arm_{\alpha}'(u) = \arm_{\alpha}(u) - \arm_{\beta}(u),
        \quad\text{and}\quad
        \arm_{s_i \permact \alpha}'(u) = \arm_{s_i \permact \alpha}(u) - \arm_{s_i \permact \beta}(u),
    \end{equation}
    which are the contributions to the arm lengths from boxes not in columns \(i\) or \(i+1\).
    This implies that \(\arm{\alpha}'(u) = \arm{s_i \permact \alpha}'(s_i \cdot u)\) for all boxes \(u \in \dg(\beta)\).
    We prove that 
    \begin{equation} \label{eq:arm-prime-descent-sum}
        \sum_{u \in \Des(T) \cap \dg(\beta)} \arm_{\alpha}'(u)
        -
        \sum_{u \in \Des(U) \cap \dg(s_i \cdot \beta)} \arm_{s_i \permact \alpha}'(u)
        = 0.
    \end{equation}

    Consider the definitions of balanced rows, left-unbalanced rows, right-unbalanced, assignment of rows to blocks in the proof of \eqref{eq:phi-maj}.
    Consider a block \(B = [r_1, r_2]\) in \(T\), and let's analyse the contributions to each side of \eqref{eq:arm-prime-descent-sum} from descents assigned to \(B\).

    If \(B\) is swapped between \(T\) and \(U\),
    then for each descent \(u\) in \(T\) assigned to \(B\), the box \(s_i \cdot u\) is a descent in \(U\) assigned to \(B\), and vice versa.
    Since \(\arm_{\alpha}'(u) = \arm_{s_i \permact \alpha}'(s_i \cdot u)\),
    the contributions to each side of \eqref{eq:arm-prime-descent-sum} from descents assigned to \(B\) are equal.

    If \(B\) is not swapped between \(T\) and \(U\) and it is not the topmost block,
    then in the proof of \eqref{eq:phi-maj}, we showed that the number of left-unbalanced rows assigned to \(B\) is equal to the number of right-unbalanced rows assigned to \(B\).
    Therefore, we can pair each left-unbalanced row assigned to \(B\) with a unique right-unbalanced row assigned to \(B\).
    By \Cref{lem:arm-cancelation},
    the contributions to each side of \eqref{eq:arm-prime-descent-sum} from each such pair are equal.

    Finally, if \(B\) is the topmost block and it is not swapped between \(T\) and \(U\),
    then in the proof of \eqref{eq:phi-maj},
    we showed that either
    \begin{itemize}
        \item the number of left-unbalanced rows assigned to \(B\) is equal to the number of right-unbalanced rows assigned to \(B\), in which case we can pair each left-unbalanced row assigned to \(B\) with a unique right-unbalanced row assigned to \(B\), and by \Cref{lem:arm-cancelation}, the contributions to each side of \eqref{eq:arm-prime-descent-sum} from each such pair are equal; or
        \item there is one extra right-unbalanced row \(r\) assigned to \(B\), the box \((i, \alpha_{i+1} + 1)\) is a descent of \(T\), and the box \((i+1, \alpha_{i+1} + 1)\) is not a descent of \(U\), in which case we check that
        \begin{multline*}
            \arm_{\alpha}'(i+1, r)
            +
            \arm_{\alpha}'(i, \alpha_{i+1} + 1)
            = \\
            \begin{aligned}
            &=
            \left|\{ j < i : \alpha_j + 1 \in \interval{r, \alpha_{i+1}} \}\right| +
            \left|\{ j > i+1 : \alpha_j \in \interval{r, \alpha_{i+1}} \}\right| \\
            &\quad+
            \left|\{ j < i : \alpha_j + 1 \in \interval{\alpha_{i+1} + 1, \alpha_i} \}\right| +
            \left|\{ j > i+1 : \alpha_j \in \interval{\alpha_{i+1} + 1, \alpha_i} \}\right| \\
            &=
            \left|\{ j < i : \alpha_j + 1 \in \interval{r, \alpha_{i+1}} \}\right| +
            \left|\{ j > i+1 : \alpha_j \in \interval{r, \alpha_{i+1}} \}\right| \\
            &=
            \arm_{s_i \permact \alpha}'(i+1, r),
            \end{aligned}
        \end{multline*}
        so the contributions to each side of \eqref{eq:arm-prime-descent-sum} from the extra left-unbalanced row \(r\) and the row \(\alpha_{i+1} + 1\) are equal; or
        \item there is one extra left-unbalanced row \(r\) assigned to \(B\), the box \((i, \alpha_{i+1} + 1)\) is not a descent of \(T\), and the box \((i+1, \alpha_{i+1} + 1)\) is a descent of \(U\), in which case we check that 
        \begin{align*}
            \arm_{\alpha}'(i, r)
            &= \left|\{ j < i : \alpha_j + 1 \in \interval{r, \alpha_{i}} \}\right| +
            \left|\{ j > i+1 : \alpha_j \in \interval{r, \alpha_{i}} \}\right| \\
            &=
            \left|\{ j < i : \alpha_j + 1 \in \interval{r, \alpha_{i+1}} \}\right| +
            \left|\{ j > i+1 : \alpha_j \in \interval{r, \alpha_{i+1}} \}\right| \\
            &\quad+
            \left|\{ j < i : \alpha_j + 1 \in \interval{\alpha_{i+1} + 1, \alpha_i} \}\right| +
            \left|\{ j > i+1 : \alpha_j \in \interval{\alpha_{i+1} + 1, \alpha_i} \}\right| \\
            &=
            \arm_{s_i \permact \alpha}'(i, r)
            + \arm_{s_i \permact \alpha}'(i+1, \alpha_{i+1} + 1),
        \end{align*}
        so the contributions to each side of \eqref{eq:arm-prime-descent-sum} from the extra left-unbalanced row \(r\) and the row \(\alpha_{i+1} + 1\) are equal.
    \end{itemize}
    In all cases, the contributions to each side of \eqref{eq:arm-prime-descent-sum} from descents assigned to \(B\) are equal.

    Therefore, we have
    \begin{align*}
        \inv(T) - \inv(U) 
        &= \left(\left|\InversionPairs(T)\right| - \left|\InversionPairs(U)\right|\right)
        - \left(\sum_{u \in \Des(T)} \arm_{\alpha}(u) - \sum_{u \in \Des(U)} \arm_{s_i \permact \alpha}(u)\right) \\
        &= \left(\left|\InversionPairs(T_1)\right| - \left|\InversionPairs(U_1)\right|\right)
        - \left(\sum_{u \in \Des(T_1)} \arm_{\beta}(u) - \sum_{u \in \Des(U_1)} \arm_{s_i \permact \beta}(u)\right) \\
        &= \inv(T_1) - \inv(U_1). 
    \end{align*}

    Finally, we check that \(\inv(T_1) = \inv(U_1)\).
    For each block \([r_1, r_2]\),
    we consider the set of type \(A\) triples
    \begin{equation*}
        \left\{
            a_r = 
            \bigl(
                (i, r), (i+1, r), (i, r-1)
            \bigr)
            : r \in \interval{r_1, r_2}
        \right\}
    \end{equation*}
    associated to the block in \(T\),
    and the set of type \(B\) triples
    \begin{equation*}
        \left\{
            b_r =
            \bigl(
                (i+1, r+1), (i, r), (i+1, r)
            \bigr)
            : r \in \interval{r_1, r_2}
        \right\}.
    \end{equation*}
    Note that this associates each triple in \(T_1\) to exactly one block, and similarly for \(U_1\).
    For each \(r \in \interval{r_1+1, r_2}\),
    since rows \(r\) and \(r-1\) are intertwined,
    the triple \(a_r\) in \(T\) has same inversion type as the triple \(b_{r+1}\) in \(U\).
    Finally, the remaining unmatched triples, \(a_{r_1}\) in \(T\) and \(b_{r_2}\) in \(U\),
    the condition to swap or not swap the block ensures that they also have the same inversion type.
    Therefore, the contributions to \(\inv(T_1)\) and \(\inv(U_1)\) from each block are equal, and consequently, \(\inv(T_1) = \inv(U_1)\).
\end{proof}


\begin{theorem}[{\cite[14]{KLO22}}] \label{thm:psi-maj-inv}
    Assume that \(\alpha_i > \alpha_{i+1}\).
    Let \(T \in \Fill(\alpha)\) and \(U = \psi(T)\).
    Let \(L = \leg(u) = \alpha_i - \alpha_{i+1} - 1\) and \(A = \arm(u)\) for \(u = (i, \alpha_{i+1} + 1) \in \dg(\alpha)\).
    If \(\ind{{T(i,1)},{T(i+1,1)}} = 1\) and \(\ind{{U(i,1)},{U(i+1,1)}} = 1\), then
    \begin{align}
        \maj(T) &= \maj(U) - (L+1),
        &&\text{and}& 
        \inv(T) &= \inv(U) + (A+1). \\
    \intertext{If \(\ind{{T(i,1)},{T(i+1,1)}} = 1\) and \(\ind{{U(i,1)},{U(i+1,1)}} = 0\), then}
        \maj(T) &= \maj(U),
        &&\text{and}& 
        \inv(T) &= \inv(U) + 1. \\
    \intertext{If \(\ind{{T(i,1)},{T(i+1,1)}} = 0\) and \(\ind{{U(i,1)},{U(i+1,1)}} = 1\), then}
        \maj(T) &= \maj(U),
        &&\text{and}&
        \inv(T) &= \inv(U) - 1. \\
    \intertext{If \(\ind{{T(i,1)},{T(i+1,1)}} = 0\) and \(\ind{{U(i,1)},{U(i+1,1)}} = 0\), then}
        \maj(T) &= \maj(U) + (L+1),
        &&\text{and}&
        \inv(T) &= \inv(U) - (A+1).
    \end{align}
\end{theorem}

\begin{proof}[Proof, independent of {\cite{KLO22}}] 
    We assume the language in the proof of \Cref{thm:phi-maj-inv}.
    Let \(T \in \Fill(\alpha)\) and \(U = \psi(T)\).
    Most of the analysis is unchanged, except for the contributions to \(\maj(T)\) and \(\inv(U)\) from the bottommost block \(B = [1, r_2]\).

    For the major statistic,
    if the bottommost block \(B\) is swapped between \(T\) and \(U\),
    then the contributions to \(\maj(T)\) and \(\maj(U)\) from descents assigned to \(B\) are equal.
    This establishes the second and third cases of the theorem.

    Assume that the bottommost block \(B = [1, r_2]\) is not swapped between \(T\) and \(U\), and that \(B\) is not the topmost block.
    Let \((a, b, c, d, e, f)\) be signature entries of \(B\).
    Since \(B\) is bottommost,
    \(f = \infty\), and thus
    \(\ind{c, d, f} = \ind{c, d} = \ind{{T(i,1)},{T(i+1,1)}} = \ind{{U(i,1)},{U(i+1,1)}}\).
    Since \(T\) and \(U\) are not swapped under \(\psi\),
    we have \(\ind{e, a, b} \neq \ind{c, d}\).

    First, assume \(\ind{a, b} = \ind{c, d}\), and consequently \(\ind{e, a} \neq \ind{e, b}\).
    As in the proof of \Cref{thm:phi-maj-inv},
    the number of left-unbalanced rows assigned to \(B\) is equal to the number of right-unbalanced rows assigned to \(B\) among rows \(2\) through \(r_2\),
    so we can pair each left-unbalanced row assigned to \(B\) with a unique right-unbalanced row assigned to \(B\).
    By \Cref{lem:leg-cancelation},
    the contributions to \(\maj(T)\) and \(\maj(U)\) from each such pair are equal.
    Moreover, we have \(\ind{e, a} \neq \ind{e, b}\),
    so row \(r_2 + 1\) is assigned to \(B\) and unbalanced to the side of the smallest of \(a\) and \(b\).
    If \(\ind{c, d} = 0\), then the smallest of \(a\) and \(b\) is \(a\),
    so row \(r_2 + 1\) is left-unbalanced, contributing \(\leg_{\alpha}(i, r_2 + 1) + 1 = \alpha_{i} - r_2\) to \(\maj(T)\) and \(\leg_{s_i \permact \alpha}(i, r_2 + 1) + 1 = \alpha_{i+1} - r_2\) to \(\maj(U)\).
    Therefore, 
    \begin{equation*}
        \maj(T) - \maj(U) = (\alpha_{i} - r_2) - (\alpha_{i+1} - r_2) = \alpha_i - \alpha_{i+1} = L + 1,
    \end{equation*}
    which agrees with the fourth case of the theorem.
    If \(\ind{c, d} = 1\), then the smallest of \(a\) and \(b\) is \(b\),
    so row \(r_2 + 1\) is right-unbalanced, contributing \(\leg_{\alpha}(i+1, r_2 + 1) + 1 = \alpha_{i+1} - r_2\) to \(\maj(T)\) and \(\leg_{s_i \permact \alpha}(i+1, r_2 + 1) + 1 = \alpha_{i} - r_2\) to \(\maj(U)\).
    Therefore, 
    \begin{equation*}
        \maj(T) - \maj(U) = (\alpha_{i+1} - r_2) - (\alpha_{i} - r_2) = \alpha_{i+1} - \alpha_i = -(L + 1),
    \end{equation*}
    which agrees with the first case of the theorem.

    Second, assume \(\ind{a, b} \neq \ind{c, d}\), and consequently \(\ind{e, a} = \ind{e, b}\).
    As in the proof of \Cref{thm:phi-maj-inv},
    \(\ind{e, a} = \ind{e, b}\) implies that row \(r_2 + 1\) is not assigned to \(B\).
    If \(\ind{c, d} = 0\) and \(\ind{a, b} = 1\),
    as in the proof of \Cref{thm:phi-maj-inv},
    there is exaclty one more left-unbalanced row than right-unbalanced rows among rows \(2\) through \(r_2\),
    so we can pair each right-unbalanced row assigned to \(B\) with a unique left-unbalanced row assigned to \(B\).
    By \Cref{lem:leg-cancelation},
    the contributions to \(\maj(T)\) and \(\maj(U)\) from each such pair are equal.
    The extra left-unbalanced row \(r\) contributes \(\leg_{\alpha}(i, r) + 1 = \alpha_{i} - r + 1\) to \(\maj(T)\) and \(\leg_{s_i \permact \alpha}(i, r) + 1 = \alpha_{i+1} - r + 1\) to \(\maj(U)\).
    Therefore, 
    \begin{equation*}
        \maj(T) - \maj(U) = (\alpha_{i} - r + 1) - (\alpha_{i+1} - r + 1) = \alpha_i - \alpha_{i+1} = L + 1,
    \end{equation*}
    which agrees with the fourth case of the theorem.
    The analysis for \(\ind{c, d} = 1\) and \(\ind{a, b} = 0\) is analogous, changing the roles of left-unbalanced and right-unbalanced rows,
    the computation for \(\maj(T) - \maj(U)\) gives \(-(L + 1)\),
    which agrees with the first case of the theorem.

    Lastly, it might be that the bottommost block \(B = [1, r_2]\) is also the topmost block.
    We omit the details, whose change from the details in the proof of \Cref{thm:phi-maj-inv} are analogous to the changes above: if \(\chi{c, d} = 1\), then there is one extra left-unbalanced row contributing \(L + 1\) to \(\maj(T) - \maj(U)\); and if \(\chi{c, d} = 0\), then there is one extra right-unbalanced row contributing \(-(L + 1)\) to \(\maj(T) - \maj(U)\).

    Now, we analyze the inversion statistic.
    As in the proof of \Cref{thm:phi-maj-inv}, \(\left|\InversionPairs(T)\right| - \left|\InversionPairs(U)\right| = \left|\InversionPairs(T_1)\right| - \left|\InversionPairs(U_1)\right|\).
    We decompose the difference \(\inv(T) - \inv(U)\) as
    \begin{gather*}
            \left(
                \left|\InversionPairs(T_1)\right|
                - \left|\InversionPairs(U_1)\right|
            \right)
            -
            \left(
                \sum_{\mathclap{\mkern60mu u \in \Des(T_1)}} \arm_{\beta}(u)
                - \sum_{\mathclap{\mkern40mu u \in \Des(U_1)}} \arm_{s_i \permact \beta}(u)
            \right)
            -
            \left(
                \sum_{\mathclap{\mkern60mu u \in \Des(T_1)}} \arm_{\alpha}'(u)
                - \sum_{\mathclap{\mkern40mu u \in \Des(U_1)}} \arm_{s_i \permact \alpha}'(u)
            \right)
            = \\
            =
            \left(\inv(T_1) - \inv(U_1)\right)
            -
            \left(
                \sum_{u \in \Des(T_1)} \arm_{\alpha}'(u)
                - \sum_{u \in \Des(U_1)} \arm_{s_i \permact \alpha}'(u)
            \right).
    \end{gather*}

    On the one hand, as in the proof of \Cref{thm:phi-maj-inv}, the number of inversion pairs in \(T_1\) assigned to each non-bottommost block is equal to the number of inversion pairs in \(U_1\) assigned to the same block.
    For the bottommost block \(B = [1, r_2]\),
    the unmatched triples, \(a_1\) in \(T\) and \(b_{r_2}\) in \(U\), have different inversion types.
    If \(\ind{c, d} = 1\), corresponding to the first and second cases of the theorem,
    then \(a_1\) is an inversion triple while \(b_{r_2}\) is not, which makes \(\left|\InversionPairs(T_1)\right| - \left|\InversionPairs(U_1)\right| = 1\).
    If \(\ind{c, d} = 0\), corresponding to the third and fourth cases of the theorem,
    then \(a_1\) is not an inversion triple while \(b_{r_2}\) is, which makes \(\left|\InversionPairs(T_1)\right| - \left|\InversionPairs(U_1)\right| = -1\).
    In summary, we have
    \begin{equation} \label{eq:inversion-pair-difference-psi}
        \left|\InversionPairs(T_1)\right| - \left|\InversionPairs(U_1)\right|
        =
        \begin{cases}
            1, & \text{if } \ind{{T(i,1)}, {T(i+1,1)}} = 1, \\
            -1, & \text{if } \ind{{T(i,1)}, {T(i+1,1)}} = 0.
        \end{cases}
    \end{equation}

    On the other hand, for each non-bottommost block, the sum of \(\arm_{\alpha}'(u)\) over descents \(u\) assigned to the block in \(T\) is equal to the sum of \(\arm_{s_i \permact \alpha}'(u)\) over descents \(u\) assigned to the block in \(U\), by the same analysis as in the proof of \Cref{thm:phi-maj-inv}.
    Therefore, it remains to analyze the contributions to the difference
    \begin{equation*}
        \sum_{u \in \Des(T_1)} \arm_{\alpha}'(u)
        -
        \sum_{u \in \Des(U_1)} \arm_{s_i \permact \alpha}'(u)
    \end{equation*}
    from descents assigned to the bottommost block \(B = [1, r_2]\).

    For the bottommost block \(B = [1, r_2]\),
    if it is swapped between \(T\) and \(U\),
    which corresponds to the second and third cases of the theorem,
    then the sum of \(\arm_{\alpha}'(u)\) over descents \(u\) assigned to \(B\) in \(T\) is equal to the sum of \(\arm_{s_i \permact \alpha}'(u)\) over descents \(u\) assigned to \(B\) in \(U\).

    Assume that the bottommost block \(B = [1, r_2]\) is not swapped between \(T\) and \(U\).
    If \(\ind{c, d} = 1\),
    which corresponds to the first case of the theorem,
    then there is one extra left-unbalanced row \(r\) assigned to \(B\) as established above.
    This extra left-unbalanced row \(r\)
    contributes \(\arm_{\alpha}'(i, r)\) to the sum over descents assigned to \(B\) in \(T\)
    and contributes \(\arm_{s_i \permact \alpha}'(i, r)\) to the sum over descents assigned to \(B\) in \(U\).
    Therefore, the difference in contributions from the extra left-unbalanced row is
    \begin{multline*}
        \arm_{\alpha}'(i, r)
        -
        \arm_{s_i \permact \alpha}'(i, r)
        = \\
        \begin{aligned}
        &=
        \left|\{ j < i : \alpha_j + 1 \in \interval{r, \alpha_i} \}\right| +
        \left|\{ j > i+1 : \alpha_j \in \interval{r, \alpha_i} \}\right| \\
        &\quad-
        \left|\{ j < i : \alpha_j + 1 \in \interval{r, \alpha_{i+1}} \}\right| -
        \left|\{ j > i+1 : \alpha_j \in \interval{r, \alpha_{i+1}} \}\right| \\
        &=
        \left|\{ j < i : \alpha_j + 1 \in \interval{\alpha_{i+1} + 1, \alpha_i} \}\right| +
        \left|\{ j > i+1 : \alpha_j \in \interval{\alpha_{i+1} + 1, \alpha_i} \}\right| \\
        &= A.
        \end{aligned}
    \end{multline*}
    If \(\ind{c, d} = 0\),
    which corresponds to the fourth case of the theorem,
    then there is one extra right-unbalanced row \(r\) assigned to \(B\) as established above.
    This extra right-unbalanced row \(r\) contributes \(\arm_{\alpha}'(i+1, r)\) to the sum over descents assigned to \(B\) in \(T\)
    and contributes \(\arm_{s_i \permact \alpha}'(i+1, r)\) to the sum over descents assigned to \(B\) in \(U\).
    Therefore, the difference in contributions from the extra right-unbalanced row is
    \begin{multline*}
        \arm_{\alpha}'(i+1, r)
        -
        \arm_{s_i \permact \alpha}'(i+1, r)
        = \\
        \begin{aligned}
        &=
        \left|\{ j < i : \alpha_j + 1 \in \interval{r, \alpha_{i+1}} \}\right| +
        \left|\{ j > i+1 : \alpha_j \in \interval{r, \alpha_{i+1}} \}\right| \\
        &\quad-
        \left|\{ j < i : \alpha_j + 1 \in \interval{r, \alpha_i} \}\right| -
        \left|\{ j > i+1 : \alpha_j \in \interval{r, \alpha_i} \}\right| \\
        &=
        - \left|\{ j < i : \alpha_j + 1 \in \interval{\alpha_{i+1} + 1, \alpha_i} \}\right| -
        \left|\{ j > i+1 : \alpha_j \in \interval{\alpha_{i+1} + 1, \alpha_i} \}\right| \\
        &= -A.
        \end{aligned}
    \end{multline*}
    In summary, we have
    \begin{equation} \label{eq:arm-prime-descent-sum-bottommost}
        \sum_{u \in \Des(T)} \arm_{\alpha}'(u)
        -
        \sum_{u \in \Des(U)} \arm_{s_i \permact \alpha}'(u)
        =
        \begin{cases}
            0, & \text{if } \ind{{T(i,1)}, {T(i+1,1)}} \neq \ind{{U(i,1)}, {U(i+1,1)}}, \\
            -A, & \text{if } \ind{{T(i,1)}, {T(i+1,1)}} = \ind{{U(i,1)}, {U(i+1,1)}} = 1, \\
            A, & \text{if } \ind{{T(i,1)}, {T(i+1,1)}} = \ind{{U(i,1)}, {U(i+1,1)}} = 0.
        \end{cases}
    \end{equation}

    Adding \eqref{eq:inversion-pair-difference-psi} and \eqref{eq:arm-prime-descent-sum-bottommost} gives the desired result.
\end{proof}

One important corollary of \Cref{thm:phi-maj-inv} is the identity~\eqref{eq:tildeH-rearrangement}, which shows that the combinatorial formula for \(\tilde{H}_{\mu}(\mathbf{x}; q, t)\) can be taken over \(\Fill[\positives](\alpha)\) for any rearrangement \(\alpha\) of the partition \(\mu\).

\begin{theorem}[{\cite{HHL08}}]
    \label{thm:tildeH-rearrangement}
    Let \(\mu\) be a partition and \(\alpha\) be a rearrangement of \(\mu\).
    Then
    \begin{equation*}
        \sum_{T \in \Fill[\positives](\mu)} q^{\maj(T)} t^{\inv(T)} \mathbf{x}^T
        =
        \sum_{T \in \Fill[\positives](\alpha)} q^{\maj(T)} t^{\inv(T)} \mathbf{x}^T,
        \tag{\ref{eq:tildeH-rearrangement}}
    \end{equation*}
    where \(\Fill[\positives](\mu)\) \textup{(}resp. \(\Fill[\positives](\alpha)\)\textup{)} is the set of fillings of shape \(\mu\) \textup{(}resp. \(\alpha\)\textup{)} with labels in the positive integers \(\positives\).
\end{theorem}

\section{Bijections on augmented fillings and a three-term identity for \texorpdfstring{\(E_\alpha^\sigma\)}{Eασ}} \label{sec:bijective-augmented}

In this section, we present bijections on augmented signed fillings to prove a three-term identity for permuted-basement Macdonald polynomials \(E_\alpha^\sigma(\mathbf{x}; q, t)\), in \cref{theorem:three-term}.

\begin{theorem}
	\label{theorem:three-term}
	Let \(\alpha\) be a composition with \(\alpha_i > \alpha_{i+1}\) for some \(i \in \interval{n-1}\).
	Then, 
	\begin{equation*}
		E_{\alpha}^{\sigma}(\mathbf{x}; q, t)
		=
		E_{s_i \permact \alpha}^{\sigma s_i}(\mathbf{x}; q, t)
		+
        \left(
        (q^{L + 1} t^{A+1})^{\ind{\sigma_{i+1}, \sigma_i}}
        t^{\rho(\alpha, \sigma, i)}
		\frac{(1 - t)}{(1 - q^{L + 1} t^{A + 1})},
        \right)
		E_{s_i \permact \alpha}^{\sigma}(\mathbf{x}; q, t).
	\end{equation*}
	where \(L = \leg(u) = \alpha_i - \alpha_{i+1} - 1\) and \(A = \arm(u)\) for \(u = (i, \alpha_{i+1} + 1) \in \dg(\alpha)\), and \(\rho(\alpha, \sigma, i) = \twinv(s_i \permact \alpha, \sigma) - \twinv(\alpha, \sigma)\).
\end{theorem}

\textcite[Proposition~17]{Ale19} proves \cref{theorem:three-term} for the case where \(\sigma_i = \sigma_{i+1} + 1\).
The specialization for \(q=0\) is proved in \cite[Proposition~5.5]{AS19}.

Our proof uses the integral form permuted-basement Macdonald polynomials \(\mathcal{E}_\alpha^\sigma\) defined in \Cref{sec:integral-form-permuted-basement}.

\begin{equation*}
    \mathcal{E}^{\sigma}_{\alpha}(\mathbf{x}; q, t) =

For the purposes of this section, let \(\wt(T) = \mathbf{x}^{|T|} q^{\maj(T)} t^{\coinv(T)} (-t)^{\operatorname{neg}(T)}\), for an augmented signed filling \(T\).

Recall from \Cref{subsec:augmented-to-non-augmented} that any augmented filling \(T \in \Fill(\alpha, \sigma)\) may be identified with a non-augmented filling \(T' \in \Fill(\alpha + \mathbf{1})\) by moving each box one unit up.
This identification preserves \(\maj\), but changes \(\coinv\).
We have \(\symcoinv(T) = \coinv(T') = \coinv(T) + \twinv(\alpha, \sigma)\).

We can apply \(\phi\) to \(T'\) to obtain \(U' = \phi(T') \in \Fill(s_i \permact \alpha + \mathbf{1})\), and then identify \(U'\) with an augmented filling \(U\) of shape \(\alpha\).
Note that \(\phi\) might or might not change the entries in positions \((i,1)\) and \((i+1,1)\) of \(T'\),
so the basement of \(U\) might be \(\sigma\) or \(\sigma s_i\).

Define the injection \(\phi \colon \Fill(\alpha, \sigma) \to \Fill(s_i \permact \alpha, \sigma) \cup \Fill(s_i \permact \alpha, \sigma s_i)\) by applying the bijection \(\phi\) to the non-augmented filling \(T'\) corresponding to \(T\), and then identifying the result with an augmented filling \(U\).
Similarly, define the injection \(\psi \colon \Fill(\alpha, \sigma) \to \Fill(s_i \permact \alpha, \sigma) \cup \Fill(s_i \permact \alpha, \sigma s_i)\) by applying the bijection \(\psi\) to the non-augmented filling \(T'\) corresponding to \(T\), and then identifying the result with an augmented filling \(U\).

We have immediate coloraries of \Cref{thm:phi-maj-inv} and \Cref{thm:psi-maj-inv}.
\begin{corollary} \label{cor:phi-symcoinv}
    For any \(T \in \Fill(\alpha, \sigma)\) and \(U = \phi(T)\), we have
    \begin{equation}
        \symcoinv(T) = \symcoinv(U)
        \quad \text{and} \quad
        \maj(T) = \maj(U).
    \end{equation}
\end{corollary}

\begin{corollary} \label{cor:psi-symcoinv}
    Assume that \(\alpha_i > \alpha_{i+1}\).
    Let \(T \in \Fill(\alpha, \sigma)\) and \(U = \psi(T)\).
    Let \(L = \leg(u) = \alpha_i - \alpha_{i+1} - 1\) and \(A = \arm(u)\) for \(u = (i, \alpha_{i+1} + 1) \in \dg(\alpha)\).
    If \(\sigma_i > \sigma_{i+1}\) and \(\pi_i > \pi_{i+1}\), then
    \begin{align}
        \maj(T) &= \maj(U) - (L+1), &&\text{and}&
        \symcoinv(T) &= \symcoinv(U) - (A+1). \\
    \intertext{If \(\sigma_i > \sigma_{i+1}\) and \(\pi_i < \pi_{i+1}\), then}
        \maj(T) &= \maj(U), &&\text{and}&
        \symcoinv(T) &= \symcoinv(U) - 1. \\
    \intertext{If \(\sigma_i < \sigma_{i+1}\) and \(\pi_i > \pi_{i+1}\), then}
        \maj(T) &= \maj(U), &&\text{and}&
        \symcoinv(T) &= \symcoinv(U) + 1. \\
    \intertext{If \(\sigma_i < \sigma_{i+1}\) and \(\pi_i < \pi_{i+1}\), then}
        \maj(T) &= \maj(U) + (L+1), &&\text{and}&
        \symcoinv(T) &= \symcoinv(U) + (A+1).
    \end{align}
\end{corollary}

\begin{definition}
    Define \(f \colon \Fill(\alpha, \sigma) \to \Fill(s_i \permact \alpha, \sigma s_i)\)
    and \(g \colon \Fill(\alpha, \sigma) \to \Fill(s_i \permact \alpha, \sigma)\) by
    \begin{equation}
        f(T) =
        \begin{cases}
            \phi(T), & \text{if } \phi(T) \in {\Fill(s_i \cdot \alpha, \sigma s_i)}, \\
            \psi(T), & \text{if } \psi(T) \in {\Fill(s_i \cdot \alpha, \sigma s_i)},
        \end{cases}
        \quad \text{and} \quad
        g(T) =
        \begin{cases}
            \phi(T), & \text{if } \phi(T) \in {\Fill(s_i \cdot \alpha, \sigma)}, \\
            \psi(T), & \text{if } \psi(T) \in {\Fill(s_i \cdot \alpha, \sigma)}.
        \end{cases}
    \end{equation}
\end{definition}

\begin{proposition}
    The maps \(f\) and \(g\) are well-defined and are bijections.
\end{proposition}

\begin{proof}
    Let \(T \in \Fill(\alpha, \sigma)\).
    Recall that the swapping conditions for \(\phi\) and \(\psi\) differ only in the bottommost block.
    Since the first row is always part of the bottommost block,
    exactly one of \(\phi(T)\) and \(\psi(T)\) has basement \(\sigma\),
    and the other has basement \(\sigma s_i\).
    Therefore, \(f\) and \(g\) are well-defined.
    The bijectivity of \(f\) and \(g\) follows from \Cref{prop:phi-inverse,prop:psi-inverse}.
\end{proof}

We split \Cref{theorem:three-term} into two cases, depending on the relative order of \(\sigma_i\) and \(\sigma_{i+1}\).
The case \(\sigma_i > \sigma_{i+1}\) is established in \Cref{thm:three-term-a-non-integral}, while the case \(\sigma_i < \sigma_{i+1}\) is established in \Cref{thm:three-term-b-non-integral}.

\subsection{Case \texorpdfstring{$\sigma_i > \sigma_{i+1}$}{σi > σi+1}}

In this subsection, we prove \cref{theorem:three-term} in the case where \(\sigma_i > \sigma_{i+1}\).
Together with \(\alpha_i > \alpha_{i+1}\), we compute that in this case
\begin{equation*}
    \twinv(\alpha, \sigma) = \twinv(s_i \permact \alpha, \sigma s_i).
\end{equation*}

\begin{lemma} \label{lem:three-term-bijective-a}
    Assume that \(\alpha_i > \alpha_{i+1}\) and \(\sigma_i > \sigma_{i+1}\).
    Let \(L = \leg(u) = \alpha_i - \alpha_{i+1} - 1\) and \(A = \arm(u)\) for \(u = (i, \alpha_{i+1} + 1) \in \dg(\alpha)\).
    Let \(T\) be an augmented filling in \(\Fill(\alpha, \sigma)\).
    Let \(U = f(T)\) and \(V = g(T)\).
    Then
    \begin{equation} \label{eq:three-term-bijective-a}
        \left( 1 - q^{L+1} t^{A + 2} \right) \wt(T)
        =
        \left( 1 - q^{L+1} t^{A + 1} \right) \wt(U)
        +
        t^{\rho(\alpha,\sigma,i)} (1 - t) \wt(V).
    \end{equation}
\end{lemma}
\begin{proof}
    First, assume \(U = \phi(T)\) and \(V = \psi(T)\).
    On the one hand, by \Cref{cor:phi-symcoinv}, \(\symcoinv(T) = \symcoinv(U)\) and \(\maj(T) = \maj(U)\).
    Since \(\twinv(\alpha, \sigma) = \twinv(s_i \permact \alpha, \sigma s_i)\),
    we have \[\coinv(T) = \symcoinv(T) - \twinv(\alpha, \sigma) = \symcoinv(U) - \twinv(s_i \permact \alpha, \sigma s_i) = \coinv(U).\]
    Thus, \(\wt(U) = \wt(T)\).
    On the other hand, by \Cref{cor:psi-symcoinv},
    we have \(\maj(T) = \maj(V) - (L+1)\) and \(\symcoinv(T) = \symcoinv(V) - (A+1)\).
    Thus,
    \begin{align*}
        \coinv(T) 
        & = \symcoinv(T) - \twinv(\alpha, \sigma) \\
        & = \symcoinv(V) - (A+1) - \twinv(\alpha, \sigma)
        = \coinv(V) - (A+1) + \rho(\alpha, \sigma, i),
    \end{align*}
    and consequently \(\wt(V) = t^{-\rho(\alpha,\sigma,i)} q^{L+1} t^{A+1} \wt(T)\).
    Therefore,
    \begin{multline*}
        \left( 1 - q^{L+1} t^{A+1} \right) \wt(U)
        +
        t^{\rho(\alpha,\sigma,i)} (1 - t) \wt(V)
        = \\
        \begin{aligned}
        &=
        \left( 1 - q^{L+1} t^{A+1} \right) \wt(T)
        +
        t^{\rho(\alpha,\sigma,i)} (1 - t) t^{-\rho(\alpha,\sigma,i)} q^{L+1} t^{A+1} \wt(T) \\
        &=
        \left( 1 - q^{L+1} t^{A+1} + (1 - t) q^{L+1} t^{A+2} \right) \wt(T) \\
        &=
        \left( 1 - q^{L+1} t^{A+2} \right) \wt(T).
        \end{aligned}
    \end{multline*}

    Second, assume \(U = \psi(T)\) and \(V = \phi(T)\).
    On the one hand, by \Cref{cor:psi-symcoinv}, \(\maj(T) = \maj(U)\) and \(\symcoinv(T) = \symcoinv(U) - 1\).
    Since \(\twinv(\alpha, \sigma) = \twinv(s_i \permact \alpha, \sigma s_i)\),
    Thus, \(\wt(U) = t \wt(T)\).
    On the other hand, by \Cref{cor:phi-symcoinv}, \(\symcoinv(T) = \symcoinv(V)\) and \(\maj(T) = \maj(V)\).
    Thus, \(\coinv(T) = \coinv(V) + \rho(\alpha, \sigma, i)\),
    and consequently \(\wt(V) = t^{-\rho(\alpha,\sigma,i)} \wt(T)\).
    Therefore,
    \begin{multline*}
        \left( 1 - q^{L+1} t^{A+1} \right) \wt(U)
        +
        t^{\rho(\alpha,\sigma,i)} (1 - t) \wt(V)
        = \\
        \begin{aligned}
        &=
        \left( 1 - q^{L+1} t^{A+1} \right) t \wt(T)
        +
        t^{\rho(\alpha,\sigma,i)} (1 - t) t^{-\rho(\alpha,\sigma,i)} \wt(T) \\
        &=
        \left( t - q^{L+1} t^{A+2} + 1 - t \right) \wt(T) \\
        &=
        \left( 1 - q^{L+1} t^{A+2} \right) \wt(T).
        \end{aligned}
    \end{multline*}
    Therefore, in either case, \eqref{eq:three-term-bijective-a} holds.
\end{proof}

\begin{corollary} \label{cor:three-term-a}
    Let \(\alpha\) be a composition such that \(\alpha_i > \alpha_{i+1}\) and \(\sigma \in S_n\) such that \(\sigma_i > \sigma_{i+1}\).
    Then,
    \begin{equation} \label{eq:three-term-a-integral}
        \left( 1 - q^{L+1} t^{A + 2} \right) \mathcal{E}_{\alpha}^\sigma
        = 
        \left( 1 - q^{L+1} t^{A + 1} \right) \mathcal{E}_{s_i \permact \alpha}^{\sigma s_i} 
        +
        t^{\rho(\alpha,\sigma,i)} (1 - t) \mathcal{E}_{s_i \permact \alpha}^\sigma,
    \end{equation}
    where \(L = \leg(u) = \alpha_i - \alpha_{i+1} - 1\) and \(A = \arm(u)\) for \(u = (i,\alpha_{i+1} +1)\in\dg(\alpha)\).
\end{corollary}

\begin{proof}
    The result follows by summing \eqref{eq:three-term-bijective-a} in \Cref{lem:three-term-bijective-a} over all \(T \in \Fill(\alpha, \sigma)\).
\end{proof}

\begin{theorem} \label{thm:three-term-a-non-integral}
    Let \(\alpha\) be a composition such that \(\alpha_i > \alpha_{i+1}\) and \(\sigma \in S_n\) such that \(\sigma_i > \sigma_{i+1}\).
    Then,
    \begin{equation} \label{eq:three-term-a-non-integral}
        E_{\alpha}^\sigma(X;q;t)
        =
        E_{s_i \permact \alpha}^{\sigma s_i}(X;q;t)
        +
        t^{\rho(\alpha,\sigma,i)}\frac{1-t}{1-q^{L+1}t^{A+1}}E_{s_i \permact \alpha}^\sigma(X;q;t)
    \end{equation}
    where \(L = \leg(u) = \alpha_i - \alpha_{i+1} - 1\) and \(A = \arm(u)\) for \(u = (i,\alpha_{i+1} +1)\in\dg(\alpha)\).
\end{theorem}

\begin{proof}
    The result follows from \Cref{lem:three-term-bijective-a} by dividing both sides of \eqref{eq:three-term-a-integral} by the appropriate factor to convert the integral forms to non-integral forms.
\end{proof}

\subsection{Case \texorpdfstring{$\sigma_i < \sigma_{i+1}$}{σi < σi+1}}

In this subsection, we prove \cref{theorem:three-term} in the case where \(\sigma_i < \sigma_{i+1}\).
Together with \(\alpha_i > \alpha_{i+1}\), we compute that in this case
\begin{equation*}
    \twinv(\alpha, \sigma) = \twinv(s_i \permact \alpha, \sigma s_i) + 1.
\end{equation*}

\begin{lemma}
    \label{lem:three-term-bijective-b}
    Assume that \(\alpha_i > \alpha_{i+1}\) and \(\sigma_i < \sigma_{i+1}\).
    Let \(L = \leg(u) = \alpha_i - \alpha_{i+1} - 1\) and \(A = \arm(u)\) for \(u = (i, \alpha_{i+1} + 1) \in \dg(\alpha)\).
    Let \(T\) be an augmented filling in \(\Fill(\alpha, \sigma)\).
    Let \(U = f(T)\) and \(V = g(T)\).
    Then
    \begin{equation} \label{eq:three-term-bijective-b}
        \left( 1 - q^{L+1} t^{A + 2} \right) \wt(T)
        =
        \left( 1 - q^{L+1} t^{A + 1} \right) \wt(U)
        +
        (q^{L+1} t^{A+1}) (1 - t) t^{\rho(\alpha,\sigma,i)} \wt(V).
    \end{equation}
\end{lemma}
\begin{proof}
    First, assume \(U = \phi(T)\) and \(V = \psi(T)\).
    On the one hand, by \Cref{cor:phi-symcoinv}, \(\maj(T) = \maj(U)\) and \(\symcoinv(T) = \symcoinv(U)\).
    Since \(\twinv(\alpha, \sigma) = \twinv(s_i \permact \alpha, \sigma s_i) + 1\),
    we have
    \begin{align*}
        \coinv(T)
        & = \symcoinv(T) - \twinv(\alpha, \sigma) \\
        & = \symcoinv(U) - \twinv(s_i \permact \alpha, \sigma s_i) - 1
        = \coinv(U) - 1.
    \end{align*}
    Thus, \(\wt(U) = t \wt(T)\).
    On the other hand, by \Cref{cor:psi-symcoinv}, we have \(\maj(T) = \maj(V) + (L+1)\) and \(\symcoinv(T) = \symcoinv(V) + (A+1)\).
    Thus,
    \begin{align*}
        \coinv(T)
        & = \symcoinv(T) - \twinv(\alpha, \sigma) \\
        & = \symcoinv(V) + (A+1) - \twinv(\alpha, \sigma)
        = \coinv(V) + (A+1) + \rho(\alpha, \sigma, i),
    \end{align*}
    and consequently \(\wt(V) = t^{-\rho(\alpha,\sigma,i)} q^{-(L+1)} t^{-(A+1)} \wt(T)\).
    Therefore,
    \begin{multline*}
        \left( 1 - q^{L+1} t^{A+1} \right) \wt(U)
        +
        (q^{L+1} t^{A+1}) (1 - t) t^{\rho(\alpha,\sigma,i)} \wt(V)
        = \\
        \begin{aligned}
        &=
        \left( 1 - q^{L+1} t^{A+1} \right) t \wt(T)
        +
        (q^{L+1} t^{A+1}) (1 - t) t^{\rho(\alpha,\sigma,i)} t^{-\rho(\alpha,\sigma,i)} q^{-(L+1)} t^{-(A+1)} \wt(T) \\
        &=
        \left( t - q^{L+1} t^{A+2} + q^{L+1} t^{A+1} - q^{L+1} t^{A+2} \right) \wt(T) \\
        &=
        \left( 1 - q^{L+1} t^{A+2} \right) \wt(T).
        \end{aligned}
    \end{multline*}

    Second, assume \(U = \psi(T)\) and \(V = \phi(T)\).
    On the one hand, by \Cref{cor:psi-symcoinv}, \(\maj(T) = \maj(U)\) and \(\symcoinv(T) = \symcoinv(U) + 1\).
    Since \(\twinv(\alpha, \sigma) = \twinv(s_i \permact \alpha, \sigma s_i) + 1\),
    \begin{align*}
        \coinv(T)
        & = \symcoinv(T) - \twinv(\alpha, \sigma) \\
        & = \symcoinv(U) + 1 - \twinv(s_i \permact \alpha, \sigma s_i) - 1
        = \coinv(U).
    \end{align*}
    Thus, \(\wt(U) = \wt(T)\).
    On the other hand, by \Cref{cor:phi-symcoinv}, \(\maj(T) = \maj(V)\) and \(\symcoinv(T) = \symcoinv(V)\).
    Thus, \(\coinv(T) = \coinv(V) + \rho(\alpha, \sigma, i)\),
    and consequently \(\wt(V) = t^{-\rho(\alpha,\sigma,i)} \wt(T)\).
    Therefore,
    \begin{multline*}
        \left( 1 - q^{L+1} t^{A+1} \right) \wt(U)
        +
        (q^{L+1} t^{A+1}) (1 - t) t^{\rho(\alpha,\sigma,i)} \wt(V)
        = \\
        \begin{aligned}
        &=
        \left( 1 - q^{L+1} t^{A+1} \right) \wt(T)
        +
        (q^{L+1} t^{A+1}) (1 - t) t^{-\rho(\alpha,\sigma,i)} t^{\rho(\alpha,\sigma,i)} \wt(T) \\
        &=
        \left( 1 - q^{L+1} t^{A+1} + (1 - t) q^{L+1} t^{A+1} \right) \wt(T) \\
        &=
        \left( 1 - q^{L+1} t^{A+2} \right) \wt(T).
        \end{aligned}
    \end{multline*}
    Therefore, in either case, \eqref{eq:three-term-bijective-b} holds.
\end{proof}

\begin{corollary} \label{cor:three-term-b}
    Let \(i \in \interval{n-1}\),
    \(\alpha = \composition{\alpha_1, \dots, \alpha_n}\) be a composition with \(\alpha_i > \alpha_{i+1}\),
    and \(\sigma = \window{\sigma_1, \dots, \sigma_n} \in \sym{n}\) be a permutation with \(\sigma_i < \sigma_{i+1}\).
    Then, we have
    \begin{equation} \label{eq:three-term-b-integral}
        \left( 1 - q^{L+1} t^{A + 2} \right) \mathcal{E}_{\alpha}^\sigma
        = 
        \left( 1 - q^{L+1} t^{A + 1} \right) \mathcal{E}_{s_i \permact \alpha}^{\sigma s_i} 
        +
        (q^{L+1} t^{A+1}) (1 - t) t^{\rho(\alpha,\sigma,i)}
        \mathcal{E}_{s_i \permact \alpha}^\sigma,
    \end{equation}
    where \(L = \leg(u) = \alpha_i - \alpha_{i+1} - 1\) and \(A = \arm(u)\) for \(u = (i,\alpha_{i+1} +1)\in\dg(\alpha)\).
\end{corollary}

\begin{proof}
    The result follows by summing \eqref{eq:three-term-bijective-b} in \Cref{lem:three-term-bijective-b} over all \(T \in \Fill(\alpha, \sigma)\).
\end{proof}

\begin{theorem} \label{thm:three-term-b-non-integral}
    Let \(\alpha\) be a composition such that \(\alpha_i > \alpha_{i+1}\) and \(\sigma \in S_n\) such that \(\sigma_i < \sigma_{i+1}\).
    Then,
    \begin{equation} \label{eq:three-term-b-non-integral}
        E_{\alpha}^\sigma(X;q;t)=E_{s_i \permact \alpha}^{\sigma s_i}(X;q;t)+(q^{L+1}t^{A+1})t^{\rho(\alpha,\sigma,i)}\frac{1-t}{1-q^{L+1}t^{A+1}}E_{s_i \permact \alpha}^\sigma(X;q;t)
    \end{equation}
    where \(L = \leg(u) = \alpha_i - \alpha_{i+1} - 1\) and \(A = \arm(u)\) for \(u = (i,\alpha_{i+1} +1)\in\dg(\alpha)\).
\end{theorem}

\begin{proof}
    The result follows from \Cref{lem:three-term-bijective-b} by dividing both sides of \eqref{eq:three-term-b-integral} by the appropriate factor to convert the integral forms to non-integral forms.
\end{proof}
