\chapter{Bijective combinatorics for Macdonald polynomials} \label{chap:bijective}

This chapter is split into two sections. \Cref{sec:bijective-non-augmented} provides bijections on non-augmented fillings to obtain formulas for the modified Macdonald polynomials \(\tilde{H}_\mu(\mathbf{x}; q, t)\).
\Cref{sec:bijective-augmented} provides bijections on augmented fillings to obtain formulas for the permuted-basement Macdonald polynomials \(E_\alpha^\sigma(\mathbf{x}; q, t)\).

In \Cref{sec:bijective-non-augmented},
we study a rearrangement property of the modified Macdonald polynomials \(\tilde{H}_\mu(\mathbf{x}; q, t)\),
\Cref{thm:tildeH-rearrangement},
which states that, for any partition \(\mu\) and any rearrangement \(\alpha\) of \(\mu\),
\begin{equation} \label{eq:tildeH-rearrangement}
    \sum_{T \in \Fill[\positives](\mu)} q^{\inv(T)} t^{\maj(T)} \mathbf{x}^T
    =
    \sum_{T \in \Fill[\positives](\alpha)} q^{\inv(T)} t^{\maj(T)} \mathbf{x}^T.
\end{equation}
The left-hand sum over \(\Fill[\positives](\mu)\) is the celebrated combinatorial expression for \(\tilde{H}_{\mu'}(\mathbf{x}; q, t)\) from \cite{HHL05} (see \Cref{subsec:modified-macdonald}), where they show that the sum satisfies the conditions of \Cref{prop:modified-macdonald-characterization}.
In \cite{HHL08}, the rearrangement property was established by proving that the right-hand sum over \(\Fill[\positives](\alpha)\) is also equal to \(\tilde{H}_{\mu'}(\mathbf{x}; q, t)\),
by showing that it also satisfies the conditions of \Cref{prop:modified-macdonald-characterization}.

Here, we present a direct bijective proof of \Cref{eq:tildeH-rearrangement}, without going through the algebraic characterization of
\(\tilde{H}_\mu(\mathbf{x}; q, t)\) from \Cref{prop:modified-macdonald-characterization}.
The key step is \Cref{thm:phi-maj-inv},
which guarantees the existence of a bijection \(\phi \colon \Fill(\alpha) \to \Fill(s_i \permact \alpha)\) that preserves \(\content\), \(\inv\), and \(\maj\).

We present two equivalent constructions of such bijection \(\phi\).
The first construction partitions the filling into blocks and decides how to transform each block based on local conditions.
The second construction is an iterative row-by-row algorithm from \cite{KLO22}, which we reformulate in a clearer way than the original presentation.
Each construction has its advantages, and we present both for completeness.
We prove that both algorithms coincide in \Cref{subsec:equivalence-of-definitions},
and then prove \Cref{thm:phi-maj-inv} in two different ways,
one for each construction.

An auxiliary bijection \(\psi \colon \Fill[\mathcal{A}](\alpha) \to \Fill[\mathcal{A}](s_i \permact \alpha)\) is also defined, that preserves \(\content\) but changes \(\inv\) and \(\maj\) in a controlled way, as specified in \Cref{thm:psi-maj-inv}.

In \Cref{sec:bijective-augmented},
we prove \Cref{thm:three-term-a-non-integral,thm:three-term-b-non-integral},
pair of three-term identities for permuted-basement Macdonald polynomials of the form
\begin{equation}
    E_\alpha^\sigma(\mathbf{x}; q, t
    =
    E_{s_i \permact \alpha}^{\sigma s_i}(\mathbf{x}; q, t)
    +
    c_{\alpha, \sigma, i}(q, t) E_{s_i \permact \alpha}^\sigma(\mathbf{x}; q, t),
\end{equation}
where \(c_{\alpha, \sigma, i}(q, t)\) is an explicit rational function in \(q\) and \(t\).

We use the same bijection \(\phi\) on non-augmented fillings (as well as an auxiliary bijection \(\psi\)) as a building block for a bijection on augmented fillings.

\section{Bijections on non-augmented fillings and the rearrangement property of \texorpdfstring{\(\tilde{H}_\mu\)}{H̃μ}} \label{sec:bijective-non-augmented}

In this section, we define the bijection \(\phi \colon \Fill(\alpha) \to \Fill(s_i \permact \alpha)\) from \cite{KLO22} that preserves the statistics \(\inv\) and \(\maj\).
As an auxiliary construction,
we also define another bijection \(\psi \colon \Fill(\alpha) \to \Fill(s_i \permact \alpha)\) that changes the statistics \(\inv\) and \(\maj\) in a controlled way, as specified in \Cref{thm:psi-maj-inv}.
This auxiliary bijection \(\psi\) is also important in \Cref{sec:bijective-augmented}.
% Without loss of generality, we restrict to the case \(\alpha_{i} > \alpha_{i+1}\), and use the inverse bijections to handle the case \(\alpha_{i} < \alpha_{i+1}\).
% \todo{Can I remove this ``without loss of generality''?}

Both of \(\phi\) and \(\psi\) follow a very rigid structure.
For now, assume \(\alpha_{i} > \alpha_{i+1}\).
If a non-augmented filling \(T \in \Fill(\alpha)\) is mapped to a filling \(U \in \Fill(s_i \permact \alpha)\) by one of these maps, then:
all columns except \(i\) and \(i+1\) are identical in \(T\) and \(U\), that is,
\begin{equation} \label{eq:restriction-other-columns}
    T(j, r) = U(j, r) \quad \text{for all } j \in \interval{n} \setminus \{i, i+1\} \text{ and all } r \in \interval{\alpha_j};
\end{equation}
the topmost \(\alpha_i - \alpha_{i+1}\) entries in column \(i\) of \(T\) are copied to the topmost \(\alpha_i - \alpha_{i+1}\) entries in column \(i+1\) of \(U\), that is,
\begin{equation} \label{eq:restriction-top-entries}
    T(i, r) = U(i+1, r) \quad \text{for all } r \in \interval{\alpha_{i+1} + 1, \alpha_i};
\end{equation}
and the entries in rows \(1, \dots, \alpha_{i+1}\) of columns \(i\) and \(i+1\) in \(T\) and \(U\) are the same, but they might be in the same positions or in swapped positions, that is,
\begin{equation} \label{eq:restriction-swap-per-row}
    \{ T(i, r), T(i+1, r) \} = \{ U(i, r), U(i+1, r) \} \quad \text{for all } r \in \interval{\alpha_{i+1}}.
\end{equation}
If \(\alpha_{i} < \alpha_{i+1}\) instead, the only difference is that the second condition swaps the roles of columns \(i\) and \(i+1\): the topmost \(\alpha_{i+1} - \alpha_{i}\) entries in column \(i+1\) of \(T\) are copied to the topmost \(\alpha_{i+1} - \alpha_{i}\) entries in column \(i\) of \(U\).
We refer to these three conditions as the \vocab{local restrictions}.

By requiring that \(\phi\) and \(\psi\) satisfy these local restrictions, the only remaining freedom in determining \(\phi\) and \(\psi\) is to decide, for each \(r \in \interval{\alpha_{i+1}}\), whether to set
\begin{multline}
    \big(U(i,r), U(i+1,r)\big) = \big(T(i,r), T(i+1,r)\big) \text{ or }\\ \big(U(i,r), U(i+1,r)\big) = \big(T(i+1,r), T(i,r)\big).
\end{multline}
In the first case, we say the entries of \(T\) and \(U\) in row \(r\) of columns \(i\) and \(i+1\) are \vocab{not swapped};
in the second case, we say they are \vocab{swapped}.
Below, we present two different algorithms to decide, for each \(r \in \interval{\alpha_{i+1}}\), whether the entries in row \(r\) of columns \(i\) and \(i+1\) are swapped or not swapped.

\subsection{Block structure algorithm} \label{subsec:block-structure}

For the block structure algorithm,
we split the rows \(\interval{\alpha_{i+1}}\) into intervals, called \emph{blocks},
and then decide, for each block, whether to swap all its rows or not swap all its rows.

To define the blocks, we define a criterion for when two consecutive rows are grouped together.
Since our goal is to construct bijections that preserve the inversion statistic \(\inv\) and inversion triples occur between two consecutive rows,
the condition we use is based on the behavior of \(\ind{{}}\) on four elements.
Before proceeding, we prove some properties of \(\ind{{}}\) when applied to four elements \(a, b, c, d \in \mathcal{A}\).

\begin{proposition} \label{prop:ind-magic}
    Let \(a, b, c, d \in \mathcal{A}\) be four elements of an alphabet \(\mathcal{A}\).
    Then,
    \begin{equation}
        \ind{a, d} + \ind{b, c} - \ind{a, c} - \ind{b, d} \in \{-1, 0, 1\}.
    \end{equation}
\end{proposition}

\begin{proof}
    Since \(\ind{x, y} \in \{0, 1\}\) for all \(x, y \in \mathcal{A}\), it's immediate that the expression only take values in \(\{-2, -1, 0, 1, 2\}\).
    For its absolute value to be \(2\), we need \(\ind{a, d} = \ind{b, c} = 1 - \ind{a, c} = 1 - \ind{b, d}\).
    This implies that \(a \geq d \geq b \geq c \geq a\) or \(a \leq d \leq b \leq c \leq a\); either way, we get \(a = b = c = d\), for which the expression evaluates to \(0\), a contradiction.
    Thus, the expression only take values in \(\{-1, 0, 1\}\).
\end{proof}

For distinct elements \(a, b, c, d \in \mathcal{A}\), the value of \(\ind{a, d} + \ind{b, c} - \ind{a, c} - \ind{b, d}\) equals \(0\) if and only if, when placing the four elements on a line in increasing order and drawing arcs between \(a\) and \(b\), and between \(c\) and \(d\), the arcs do not cross.
See \Cref{fig:ind-magic} for two examples.
We have \Cref{prop:even-ind-magic,prop:odd-ind-magic} that give some other equivalent characterizations of whether the expression is \(0\) or \(\pm 1\), which work even when the elements are not distinct.

\begin{proposition} \label{prop:even-ind-magic}
    Let \(a, b, c, d \in \mathcal{A}\).
    The following statements are equivalent:
    \begin{enumerate}
        \item \label{item:even-ind}
        \(\ind{a, d} + \ind{b, c} - \ind{a, c} - \ind{b, d} = 0\);
        \item \label{item:ind-abx}
        \(\ind{a, b, c} = \ind{a, b, d}\);
        \item \label{item:ind-xcd}
        \(\ind{a, c, d} = \ind{b, c, d}\).
    \end{enumerate}
\end{proposition}

\begin{proof}
    Expanding the definition of \(\ind{x, y, z}\), \Cref{item:ind-abx} is equivalent to
    \(\ind{a, b} + \ind{b, c} - \ind{a, c} = \ind{a, b} + \ind{b, d} - \ind{a, d}\),
    which rearranges to \Cref{item:even-ind}.
    The equivalence between \Cref{item:ind-xcd} and \Cref{item:even-ind} is analogous.
\end{proof}

\begin{proposition} \label{prop:odd-ind-magic}
    Let \(a, b, c, d \in \mathcal{A}\).
    The following statements are equivalent to the negation of those in \Cref{prop:even-ind-magic}:
    \begin{enumerate}[start=4]
        \item \label{item:odd-ind}
        \(\ind{a, d} + \ind{b, c} - \ind{a, c} - \ind{b, d} = \pm 1\);
        \item \label{item:odd-ind-2}
        \(\ind{a, d} + \ind{b, c} + \ind{a, c} + \ind{b, d}\) is odd;
        \item \label{item:triple-equal-ind}
        \(\ind{a, b, c} = \ind{b, c, d} = \ind{a, d, c}\) and \(a \neq b\) and \(c \neq d\);
    \end{enumerate}
\end{proposition}

\begin{proof}
    The equivalence between \Cref{item:odd-ind} and the negation of \Cref{item:even-ind} is immediate from \Cref{prop:ind-magic}, as well as the equivalence between \Cref{item:odd-ind} and \Cref{item:odd-ind-2} using modulo \(2\).
    Computational verification shows that \Cref{item:odd-ind} and \Cref{item:triple-equal-ind}
    are satisfied for exactly the same relative configurations of \(a, b, c, d\), namely one of the following \(40\) configurations:

    a \overset{-}{=} c > b \overset{-}{=} d, \quad

    b \overset{+}{=} c > a \overset{+}{=} d, \quad
    b \overset{-}{=} c > a \overset{-}{=} d, \quad

    a > d > b \overset{+}{=} c, \\[4pt]

    c > b > a \overset{-}{=} d, \quad

Let \(r \in \interval{\alpha_{i+1} - 1}\).
Let \( a = T(i,r+1)\), \(b = T(i+1,r+1)\), \(c = T(i,r)\), and \(d = T(i+1,r)\).
We say that rows \(r\) and \(r + 1\) of \(T\) are \vocab{not intertwined} if \(a, b, c, d\) satisfy the equivalent conditions of \Cref{prop:even-ind-magic}, and \vocab{intertwined} if they satisfy the equivalent conditions of \Cref{prop:odd-ind-magic}.
Note that the condition is invariant under swapping \(a\) with \(b\), as well as invariant under swapping \(c\) with \(d\).

The interval of rows \(\interval{\alpha_{i+1}}\) is uniquely partitioned into intervals, called \vocab{blocks}, such that two consecutive rows in the same block are intertwined, and two consecutive rows in different blocks are not intertwined.
The partition of \(\interval{\alpha_{i+1}}\) into blocks is called the \vocab{block structure (of columns \(i\) and \(i+1\)) of the filling \(T\)}.
\Cref{fig:blocks} shows a filling and its blocks.

Note that, if the two entries of row \(r\) of columns \(i\) and \(i+1\) in \(T\) are equal, then rows \(r - 1\) and \(r\) are not intertwined, and rows \(r\) and \(r + 1\) are not intertwined (if these rows exist).
Therefore, such row \(r\) forms a block of size \(1\) by itself, and swapping or not swapping it has the same effect.

Let's interpret the intertwining condition via~\Cref{item:triple-equal-ind} of \Cref{prop:odd-ind-magic}.
The number \(\ind{a, b, c}\) is the indicator of whether the triple \((i, r+1), (i+1, r+1), (i, r)\) is a Type A inversion triple in \(T\).
The number \(\ind{b, c, d}\) is the indicator of whether the triple \((i+1, r+1), (i, r), (i+1, r)\) is a Type B inversion triple in \(U\) if rows \(r\) and \(r + 1\) are not swapped.
The number \(\ind{a, d, c}\) is the indicator of whether the triple \((i, r+1), (i+1, r), (i, r)\) is a Type B inversion triple in \(U\) if rows \(r\) and \(r + 1\) are swapped.

Therefore,
if we require that rows in the same block are either all swapped or all not swapped in \(U\),
then we guarantee that the number of inversion triples within each block is the same in \(T\) and \(U\), regardless of whether the block is swapped or not swapped.

The remaining freedom is to decide, for each block, whether to swap it or not swap it.
We use this freedom to guarantee that the number of inversion triples between different blocks is also the same in \(T\) and \(U\).

Consider a block \([r_1, r_2]\) of \(T\).
Let
\begin{align*}
    e_1 &= T(i, r_2 + 1),  & e_2 &= T(i+1, r_2 + 1), \\
    a   &= T(i, r_2),      & b   &= T(i+1, r_2),     \\
    c   &= T(i, r_1),      & d   &= T(i+1, r_1),     \\
    f_1 &= T(i, r_1 - 1), \text{ and} & f_2 &= T(i+1, r_1 - 1).
\end{align*}
Edge cases are when the block is the bottommost one, i.e. \(r_1 = 1\),
in which case \(f_1 = f_2 = \infty\) by convention;
and when the block is the topmost one, i.e. \(r_2 = \alpha_{i+1}\), in which case one of \(e_1, e_2\) is undefined.
For any well-defined \(e \in \{e_1, e_2\}\) and \(f \in \{f_1, f_2\}\),
we say that \((a, b, c, d, e, f)\) are \vocab{signature entries} of the block.
See \Cref{fig:general-block} for an illustration of the entries \(a, b, c, d, e_1, e_2, f_1, f_2\).

If the block is not the bottommost one, then rows \(r_1 - 1\) and \(r_1\) are not intertwined.
Therefore, \(c, d, f_1, f_2\) satisfy the equivalent conditions of \Cref{prop:even-ind-magic}, in particular, \(\ind{c, d, f_1} = \ind{c, d, f_2}\).
If the block is the bottommost one,
then we have \(\ind{c, d, f_1} = \ind{c, d, f_2} = \ind{c, d}\) by convention (as expected from \(f_1 = f_2 = \infty\)).
Therefore, in all cases, the value \(\ind{c, d, f}\) doesn't depend on the choice of \(f \in \{f_1, f_2\}\).

If the block is not the topmost one, then rows \(r_2\) and \(r_2 + 1\) are not intertwined.
Therefore, \(a, b, e_1, e_2\) satisfy the equivalent conditions of \Cref{prop:even-ind-magic}, in particular, \(\ind{e_1, a, b} = \ind{e_2, a, b}\).
If the block is the topmost one, then exactly one of \(e_1\) and \(e_2\) is defined, so there is no ambiguity in choosing \(e\) and in computing \(\ind{e, a, b}\).
Therefore, in all cases, the value \(\ind{e, a, b}\) doesn't depend on the choice of \(e \in \{e_1, e_2\}\) (as long as \(e\) is defined).

In all cases, the values \(\ind{c, d, f}\) and \(\ind{e, a, b}\) are uniquely defined for a block, independently of the choice of signature entries \((a, b, c, d, e, f)\).

\begin{definition}[The bijection \(\phi\) via blocks] \label{def:phi-via-blocks}
    Let \(T \in \Fill(\alpha)\).
    We define \(U = \phi(T) \in \Fill(s_i \permact \alpha)\)
    by requiring the local restrictions (i.e., \Cref{eq:restriction-other-columns,eq:restriction-top-entries,eq:restriction-swap-per-row})
    and requiring that, for each block of \(T\) with signature entries \((a, b, c, d, e, f)\),
    the rows of the block are swapped in \(U\) if and only if
    \begin{equation}
        \ind{c, d, f} \neq \ind{e, a, b}.
    \end{equation}
\end{definition}

\begin{definition}[The bijection \(\psi\) via blocks] \label{def:psi-via-blocks}
    Let \(T \in \Fill(\alpha)\).
    We define \(U = \psi(T) \in \Fill(s_i \permact \alpha)\)
    by requiring the local restrictions (i.e., \Cref{eq:restriction-other-columns,eq:restriction-top-entries,eq:restriction-swap-per-row})
    and requiring that, for each block of \(T\) with signature entries \((a, b, c, d, e, f)\),
    the rows of the block are swapped in \(U\) if and only if
    \begin{equation}
        \ind{c, d, f} \neq \ind{e, a, b} \quad \text{xor} \quad \text{(the block is the bottommost block)}.
    \end{equation}
\end{definition}

To make things explicitly, both \(\phi\) and \(\psi\) swap a non-bottommost block if and only if \(\ind{c, d, f} \neq \ind{e, a, b}\).
The difference is in the bottommost block, which \(\phi\) swaps if and only if \(\ind{c, d} \neq \ind{e, a, b}\),
while \(\psi\) swaps it if and only if \(\ind{c, d} = \ind{e, a, b}\).

In \Cref{prop:phi-inverse,prop:psi-inverse,thm:phi-maj-inv}, we prove that \(\phi\) is a bijection that preserves \(\content\), \(\inv\), and \(\maj\).
Let's highlight the parts that are straightforward from the definition via blocks.

We verify that both \(\phi\) and \(\psi\) are bijections by noticing that \(\phi(\phi(T)) = T\) and \(\psi(\psi(T)) = T\).
The local restrictions guarantee that \(\content(T) = \content(U)\).

Moreover, it is straightforward to verify that the number of inversion triples within columns \(i\) and \(i+1\) is the same in \(T\) and \(\phi(T)\), since the number of inversion triples within each block is preserved, and the choice of swapping or not swapping each block is made precisely to ensure that the number of inversion triples between different blocks is also preserved.
It is not hard to see that this implies that \(\inv(T) = \inv(\phi(T))\).

The fact that \(\maj(T) = \maj(\phi(T))\) is less straightforward,
since we haven't made any direct attempt to control the major index in our definition of \(\phi\).
We prove this in \Cref{thm:phi-maj-inv}.

\subsection{Row-by-row algorithm} \label{subsec:row-by-row}
It turns out that the bijections \(\phi\) and \(\psi\) are the same as the bijections defined in \cite{KLO22}.
We now present the algorithm from \cite{KLO22}, in a clearer form than the original exposition.

Let \(T \in \Fill(\alpha)\). Let \(h = \min(\alpha_i, \alpha_{i+1})\).
We describe the algorithm to compute \(U = \phi(T)\) or \(U = \psi(T)\).
The algorithm uses a \vocab{coin} with two sides, \(\coinphi\) and \(\coinpsi\), as a bookkeeping device.
Each row from \(1\) to \(\alpha_{i+1} - 1\) is, at some point of the algorithm, assigned a \vocab{coin value} in \(\{\coinphi, \coinpsi\}\).
Initially, assign the coin value of row \(1\) to be \(\coinphi\) if computing \(U = \phi(T)\), or \(\coinpsi\) if computing \(U = \psi(T)\).
We denote by \(\operatorname{coin}_r\) the coin value assigned to row \(r\).

In addition to the coin value, each row from \(1\) to \(\alpha_{i+1}\) is labeled from \(\{\labelN, \labelS, \labelC\}\).
A label of \(\labelN\) means the entries in row \(r\) of columns \(i\) and \(i+1\) in \(T\) and \(U\) are \emph{not} swapped;
a label of \(\labelS\) means they are \emph{swapped};
and a label of \(\labelC\) indicates that the behavior for row \(r\) should \emph{copy} the behavior for the row above it.
We denote by \(\operatorname{label}_r\) the label assigned to row \(r\).

We process the rows iteratively from row \(1\) up to row \(h\).
For each \(r \in \{1,2,\dots, h-1\}\),
we assume that the coin value \(\operatorname{coin}_r\) has already been assigned.
Let \( a = T(i,r+1)\), \(b = T(i+1,r+1)\), \(c = T(i,r)\), and \(d = T(i+1,r)\).
Note that \(\ind{a, d} = \ind{b, c} \neq \ind{a, c} = \ind{b, d}\) is impossible by transitivity,
thus \( \ind{a,d} + \ind{b,c} - \ind{a,c} - \ind{b,d} \in \{1, 0, -1\} \).

\begin{description}
    \item[Case I.] If 
    \(\ind{a,d} + \ind{b,c} - \ind{a,c} - \ind{b,d} = \pm 1\),
    (cases 3, 4, 9, 10 in \cite{KLO22}),  
    then set \(\operatorname{label}_r = \labelC\).
    Moreover, set \(\operatorname{coin}_{r+1} = \operatorname{coin}_r\) if \(\chi_{ab} = \chi_{cd}\) (cases 3, 9 in \cite{KLO22}),
    and set \(\operatorname{coin}_{r+1} \neq \operatorname{coin}_r\) if \(\chi_{ab} \neq \chi_{cd}\) (cases 4, 10 in \cite{KLO22}).

    \item[Case II.] If
    \(\ind{a,c} = \ind{a,d} = \ind{b,c} = \ind{b,d}\)
    (cases 1, 2, 11, 12 in \cite{KLO22}),
    then set \(\operatorname{label}_r = \labelN\) if \(\operatorname{coin}_r = \coinphi\), and set \(\operatorname{label}_r = \labelS\) if \(\operatorname{coin}_r = \coinpsi\).
    Moreover, set \(\operatorname{coin}_{r+1} = \coinphi\).
    
    \item[Case III.] If 
    \(\ind{a,c} = \ind{b,c} \neq \ind{a,d} = \ind{b,d}\)  
    (cases 7, 8 in \cite{KLO22}),
    then set \(\operatorname{label}_r = \labelS\) if \(\operatorname{coin}_r = \coinphi\), and set \(\operatorname{label}_r = \labelN\) if \(\operatorname{coin}_r = \coinpsi\).
    Moreover, set \(\operatorname{coin}_{r+1} = \coinphi\).
    
    \item[Case IV.] If
    \(\ind{a,c} = \ind{a,d} \neq \ind{b,c} = \ind{b,d}\)
    (cases 5, 6 in \cite{KLO22}),
    then set \(\operatorname{label}_r = \labelN\) if \(\operatorname{coin}_r = \coinphi\), and set \(\operatorname{label}_r = \labelS\) if \(\operatorname{coin}_r = \coinpsi\).
    Moreover, set \(\operatorname{coin}_{r+1} = \coinpsi\).
\end{description}

Finally, we process row \(r = h\), assuming that the coin value \(\operatorname{coin}_h\) has already been assigned.
Let \(a = T(i,h+1)\) if \(h = \alpha_{i+1} < \alpha_i\) or \(a = T(i+1,h+1)\) if \(h = \alpha_{i} < \alpha_{i+1}\), i.e., \(a\) is the only entry in row \(h+1\) on one of the two columns \(i\) or \(i+1\).
Let \(c = T(i,h)\), and let \(d = T(i+1,h)\).

\begin{description}
    \item[Base I.] If \(\ind{a,c} \neq \ind{a,d}\),
    then set \(\operatorname{label}_r = \labelS\) if \(\operatorname{coin}_r = \coinphi\), and set \(\operatorname{label}_r = \labelN\) if \(\operatorname{coin}_r = \coinpsi\).

    \item[Base II.] If \(\ind{a,c} = \ind{a,d}\), then set \(\operatorname{label}_r = \labelN\) if \(\operatorname{coin}_r = \coinphi\), and set \(\operatorname{label}_r = \labelS\) if \(\operatorname{coin}_r = \coinpsi\).
\end{description}

\begin{definition}[The bijections \(\phi\) and \(\psi\) via the row-by-row algorithm] \label{def:phi-psi-via-row-by-row}

    The row-by-row algorithm described above, with initial coin value \(\coinphi\) (resp.~\(\coinpsi\)), produces a label for each row \(r \in \interval{\alpha_{i+1}}\).
    We define \(U = \phi(T) \in \Fill(s_i \permact \alpha)\) (resp.~\(U = \psi(T) \in \Fill(s_i \permact \alpha)\))
    by requiring the local restrictions (i.e., \Cref{eq:restriction-other-columns,eq:restriction-top-entries,eq:restriction-swap-per-row}) and requiring that, for each row \(r \in \interval{\alpha_{i+1}}\),

The row-by-row algorithm may be described recursively instead of iteratively.

Let \(T \in \Fill(\alpha)\) with \(h = \min(\alpha_i, \alpha_{i+1})\).
Assign the coin value of row \(1\) to be \(\coinphi\) if computing \(U = \phi(T)\), or \(\coinpsi\) if computing \(U = \psi(T)\).
If column \(i+1\) consists of a single entry, then assign the label of this row according to Base~I or~II.
Otherwise, assign the label of row \(1\) according one of Cases~I--IV,
and at the same time determine the coin value to be used for the next row.
Remove the first row to obtain the sub-filling \(T'\), and apply the algorithm recursively to \(T'\) with this new coin value, obtaining the labels for rows \(2,\dots, h\).
Once all labels are determined, construct \(U\) in a final pass by applying the rules for \(\labelN,\labelS,\labelC\) as before.

It is straightforward to verify that both the iterative and recursive descriptions of the algorithm are equivalent.
This recursive interpretation is particularly useful for proving properties about the bijections \(\phi\) and \(\psi\), as it allows us to use induction on the number of rows in the filling.

\subsection{Equivalence of the two definitions} \label{subsec:equivalence-of-definitions}

We now show that the two definitions of \(\phi\) and \(\psi\) given in \Cref{subsec:block-structure,subsec:row-by-row} are equivalent.
We do so by applying the row-by-row algorithm and observing what happens to each block.

\begin{proposition} \label{prop:equivalence-of-definitions}
    The definitions of \(\phi\) and \(\psi\) via blocks (\Cref{def:phi-via-blocks,def:psi-via-blocks}) are equivalent to the definitions via the row-by-row algorithm (\Cref{def:phi-psi-via-row-by-row}).
\end{proposition}

\begin{proof}
    Let \(\phi\) be defined as in \Cref{def:phi-psi-via-row-by-row}.
    We show that \(\phi\) satisfies the conditions in \Cref{def:phi-via-blocks}.
    Without loss of generality, assume \(\alpha_i > \alpha_{i+1}\).
    Let \(T \in \Fill(\alpha)\).
    Consider a block \([r_1, r_2]\) of \(T\).
    Let \(e_1 = T(i, r_2 + 1)\), \(a = T(i, r_2)\), \(b = T(i+1, r_2)\), \(c = T(i, r_1)\), \(d = T(i+1, r_1)\), and \(f_1 = T(i, r_1 - 1)\),
    so that \((a, b, c, d, e_1, f_1)\) are signature entries of the block.

    First, if \(r_1 > 1\),
    consider the row-by-row algorithm at row \(r_1 - 1\), and let's compute the coin value for row \(r_1\).
    Since \(r_1 - 1\) and \(r_1\) are not intertwined,
    we are in Case~II, III, or~IV.
    Therefore, we have
    \begin{equation} \label{eq:coin-r1}
        \operatorname{coin}_{r_1} = \coinphi
        \quad \text{if and only if} \quad
        \ind{c, f_1} - \ind{d, f_1} = 0.
    \end{equation}
    If \(r_1 = 1\), then by definition, \(\operatorname{coin}_{r_1} = \coinphi\),
    and by convention, \(\ind{c, f_1} - \ind{d, f_1} = 0\) since \(f_1 = \infty\).
    Thus, \Cref{eq:coin-r1} holds independently of whether \(r_1 = 1\) or \(r_1 > 1\).

    Second, for each \(r \in \{r_1, r_1 + 1, \dots, r_2 - 1\}\),
    since rows \(r\) and \(r + 1\) are intertwined,
    the row-by-row algorithm falls into Case~I, \(\operatorname{label}_r = \labelC\).
    Thus, the rows in the block all copy the behavior of row \(r_2\) and the label of row \(r_2\) determines whether the block is swapped or not swapped,
    that is, the entire block is swapped if and only if \(\operatorname{label}_{r_2} = \labelS\).
    
    Moreover, \(\operatorname{coin}_{r+1} = \operatorname{coin}_r\) if and only if \(\ind{{T(i,r+1)},{T(i+1,r+1)}} = \ind{{T(i,r)},{T(i+1,r)}}\).
    Stringing these together, we have
    \begin{equation} \label{eq:coin-r2-coin-r1}
        \operatorname{coin}_{r_2} = \operatorname{coin}_{r_1}
        \quad \text{if and only if} \quad
        \ind{a, b} - \ind{c, d} = 0.
    \end{equation}

    Third, consider the row-by-row algorithm at row \(r_2\), and let's compute the label for row \(r_2\).
    Since \(r_2\) and \(r_2 + 1\) are not intertwined,
    we are in Case~II, III, or~IV or Base~I or~II.
    In all of these cases, we have
    \begin{equation} \label{eq:label-r2}
        \operatorname{label}_{r_2} = \labelS
        \quad \text{if and only if} \quad
        (\ind{e_1, a} - \ind{e_1, b} = 0) \mathbin{\text{xor}} (\operatorname{coin}_{r_2} = \coinpsi).
    \end{equation}

    Finally, using that \(\operatorname{coin}_{r_2} = \coinpsi\) if and only if \( (\operatorname{coin}_{r_2} = \operatorname{coin}_{r_1} ) \mathbin{\text{xor}} (\operatorname{coin}_{r_1} = \coinphi) \),
    we have
    \begin{align*}
        \operatorname{label}_{r_2} = \labelS
        &\iff (\ind{e_1, a} - \ind{e_1, b} = 0) \mathbin{\text{xor}} (\operatorname{coin}_{r_2} = \operatorname{coin}_{r_1} ) \mathbin{\text{xor}} (\operatorname{coin}_{r_1} = \coinphi) \\
        &\iff (\ind{e_1, a} - \ind{e_1, b} = 0) \mathbin{\text{xor}} (\ind{a, b} - \ind{c, d} = 0) \mathbin{\text{xor}} (\ind{c, f_1} - \ind{d, f_1} = 0) \\
        &\iff \ind{e_1, a} - \ind{e_1, b} + \ind{a, b} - \ind{c, d} + \ind{c, f_1} - \ind{d, f_1} \equiv 1 \pmod{2} \\
        &\iff \ind{e_1, a, b} \neq \ind{c, d, f_1}.
    \end{align*}
    which is the same swapping condition as in \Cref{def:phi-via-blocks}.
    Therefore, the two definitions of \(\phi\) are equivalent.

    The analysis for \(\psi\) is analogous.
    For \(\psi\), \Cref{eq:coin-r1} becomes
    \begin{equation}
        \operatorname{coin}_{r_1} = \coinphi
        \quad \text{if and only if} \quad
        (\ind{c, f_1} - \ind{d, f_1} = 0) \mathrel{\text{xor}} (r_1 = 1),
    \end{equation}
    while \Cref{eq:coin-r2-coin-r1,eq:label-r2} remain unchanged.
    Therefore, the conclusion becomes
    \begin{equation*}
        \operatorname{label}_{r_2} = \labelS
        \iff (\ind{e_1, a, b} \neq \ind{c, d, f_1}) \mathbin{\text{xor}} (r_1 = 1),
    \end{equation*}
    which is the same swapping condition as in \Cref{def:psi-via-blocks}.
    Therefore, the two definitions of \(\psi\) are equivalent.
\end{proof}

\subsection{Properties of the bijections \texorpdfstring{\(\phi\)}{𝜙} and \texorpdfstring{\(\psi\)}{𝜓}}

As an abuse of notation, let \(\phi\) denote both the function \(\phi \colon \Fill(\alpha) \to \Fill(s_i \permact \alpha)\) and the function \(\phi \colon \Fill(s_i \permact \alpha) \to \Fill(\alpha)\).
The intended use of \(\phi\) is always clear from the shape of the input filling.

\begin{proposition} \label{prop:phi-inverse}
    The maps \(\phi \colon \Fill(\alpha) \to \Fill(s_i \permact \alpha)\) and \(\phi \colon \Fill(s_i \permact \alpha) \to \Fill(\alpha)\) are inverses of each other.
    Moreover, \(\phi\) is a bijection.
\end{proposition}

\begin{proof}[Proof (via block structure)]
    Let \(T \in \Fill(\alpha)\).
    Let \(U = \phi(T)\) and \(T' = \phi(U)\).
    Since the condition of two rows being intertwined only depends on the set of entries in two rows, not the order of the entries,
    the block structure of \(T\), \(U\), and \(T'\) are the same.
    Let \([r_1, r_2]\) be a block of \(T \in \Fill(\alpha)\) with signature entries \((a, b, c, d, e, f)\).

    If \(a = b\) or \(c = d\), then the block consists of a single row, since a row of equal entries cannot be intertwined with any other row.
    Therefore, the columns \(i\) and \(i+1\) of this row are identical in \(T\), \(U\), and \(T'\).
    Thus, we may assume that \(a \neq b\) and \(c \neq d\).

    By~\Cref{def:phi-via-blocks}, the rows in \([r_1, r_2]\) are swapped between \(T\) and \(U\) if and only if \( \ind{c, d, f} \neq \ind{e, a, b} \).

    If this condition is false, then the rows in \([r_1, r_2]\) are not swapped between \(T\) and \(U\).
    Therefore, \([r_1, r_2]\) is a block of \(U\) with signature entries \((a, b, c, d, e, f)\), irregardless of whether the blocks immediately above or below are swapped or not.
    Now, again by~\Cref{def:phi-via-blocks}, the rows in \([r_1, r_2]\) are swapped between \(U\) and \(T'\) if and only if \(\ind{c, d, f} \neq \ind{e, a, b}\), which is false.
    Therefore, the rows in \([r_1, r_2]\) are not swapped between \(U\) and \(T'\), and consequently, \(T\) and \(T'\) are identical in this block.

    If this condition is true, then the rows in \([r_1, r_2]\) are swapped between \(T\) and \(U\).
    Therefore, \([r_1, r_2]\) is a block of \(U\) with signature entries \((b, a, d, c, e, f)\), irregardless of whether the blocks immediately above or below are swapped or not.
    Now, by~\Cref{def:phi-via-blocks}, the rows in \([r_1, r_2]\) are swapped between \(U\) and \(T'\) if and only if \(\ind{d, c, f} \neq \ind{e, b, a}\).
    Because \(a \neq b\) and \(c \neq d\), we have \(\ind{d, c, f} = 1 - \ind{c, d, f}\) and \(\ind{e, b, a} = 1 - \ind{e, a, b}\).
    Therefore, \(\ind{d, c, f} \neq \ind{e, b, a}\) is equivalent to \(\ind{c, d, f} \neq \ind{e, a, b}\), which is true.
    Thus, the rows in \([r_1, r_2]\) are swapped between \(U\) and \(T'\).
    Since the rows in \([r_1, r_2]\) are also swapped between \(T\) and \(U\), then \(T\) and \(T'\) are identical in this block.

    Therefore, for all blocks of \(T\), the corresponding blocks of \(T'\) are identical.
    All other entries of \(T\) and \(T'\) outside of blocks are equal by the local restrictions.
    Thus, \(T = T'\), and consequently \(\phi \colon \Fill(\alpha) \to \Fill(s_i \permact \alpha)\) is the inverse of \(\phi \colon \Fill(s_i \permact \alpha) \to \Fill(\alpha)\).
    Since \(\phi\) has an inverse, it is a bijection.
\end{proof}

\begin{proposition} \label{prop:psi-inverse}
    The maps \(\psi \colon \Fill(\alpha) \to \Fill(s_i \permact \alpha)\) and \(\psi \colon \Fill(s_i \permact \alpha) \to \Fill(\alpha)\) are inverses of each other.
    Moreover, \(\psi\) is a bijection.
\end{proposition}

The proof of \Cref{prop:psi-inverse} is analogous to the proof of \Cref{prop:phi-inverse}, with a minor modification to account for the different swapping condition of the bottommost block.

\begin{theorem}[{\cite[Proposition~4.3]{KLO22}}] \label{thm:phi-maj-inv}
    The bijection \(\phi\) preserves the statistics \(\inv\) and \(\maj\); that is, for all \(T \in \Fill(\alpha)\), if \(U = \phi(T)\), then
    \begin{equation}
        \inv(T) = \inv(U) \quad \text{and} \quad \maj(T) = \maj(U).
    \end{equation}
\end{theorem}

\begin{proof}[Proof (via block structure)]
    Without loss of generality, assume \(\alpha_i > \alpha_{i+1}\).
    Let \(T \in \Fill(\alpha)\) and \(U = \phi(T)\).
    We first study how \(\Des(T)\) and \(\Des(U)\) relate to each other.
    Given a box \(u = (j, r)\), let \(s_i \cdot u = (s_i(j), r)\) be the box obtained by applying the transposition \(s_i\) to the column index of \(u\).

    Clearly, for all boxes \(u\) not in columns \(i\) or \(i+1\), we have \(u \in \Des(T)\) if and only if \(u \in \Des(U)\), by \Cref{eq:restriction-other-columns}.
    Since \(\leg_\alpha(u) = \leg_{s_i \permact \alpha}(u)\) for such boxes \(u\),
    these descents contribute equally to \(\maj(T)\) and \(\maj(U)\).
    Moreover, for all boxes \(u = (i, r)\) with \(r \geq \alpha_{i+1} + 2\),
    we have \(u \in \Des(T)\) if and only if \(s_i \cdot u \in \Des(U)\), by \Cref{eq:restriction-top-entries}.
    Note that \(\leg_\alpha(u) = \alpha_{i} - r = \leg_{s_i \permact \alpha}(s_i \cdot u)\) for such boxes \(u\),
    so these descents contribute equally to \(\maj(T)\) and \(\maj(U)\).
    Therefore, it remains to understand the descents occurring in columns \(i\) and \(i+1\) at rows \(\interval{2, \alpha_{i+1} + 1}\).

    For any row \(r \in \interval{\alpha_{i+1}}\), let \(\ind{r} = \ind{T(i,r), T(i+1,r)}\).
    If both or neither of the boxes \((i, r)\) and \((i+1, r)\) are descents of \(T\), we say that row \(r\) is \vocab{balanced} in \(T\);
    if only \((i, r)\) is a descent of \(T\), we say that row \(r\) is \vocab{unbalanced to the left} in \(T\);
    and if only \((i+1, r)\) is a descent of \(T\), we say that row \(r\) is \vocab{unbalanced to the right} in \(T\).

    Each of the rows in \([2, \alpha_{i+1} + 1]\) will be \vocab{assigned} to a unique block of \(T\), which is not necessarily the block that contains that row.
    We describe how the assignment is done.

    As an edge case, row \(\alpha_{i+1} + 1\) is assigned to the topmost block (i.e., the block containing row \(\alpha_{i+1}\)).

    Let \(r \in \interval{\alpha_{i+1}-1}\).
    Let \(a = T(i, r+1)\), \(b = T(i+1, r+1)\), \(c = T(i, r)\), and \(d = T(i+1, r)\).
    
    First, assume rows \(r\) and \(r+1\) are not intertwined in \(T\), that is \(\ind{a, c} + \ind{b, d} = \ind{a, d} + \ind{b, c}\).
    This means that \(r\) is the highest row of a block \(B_1\),
    and \(r+1\) is the lowest row of the block immediately above \(B_1\), say \(B_2\).
    If \(\ind{a, c} = \ind{b, d} = 1\), meaning that both \((i, r+1)\) and \((i + 1, r + 1)\) are descents of \(T\),
    then we also have \(\ind{a, d} = \ind{b, c} = 1\), so independently of whether each of the blocks is swapped or not, both \((i, r+1)\) and \((i + 1, r + 1)\) are descents of \(U\) as well.
    In this case, we \vocab{assign} these descents to the block \(B_1\).
    Similarly, if \(\ind{a, c} = \ind{b, d} = 0\), meaning that neither \((i, r+1)\) nor \((i + 1, r + 1)\) is a descent of \(T\),
    then we also have \(\ind{a, d} = \ind{b, c} = 0\), so independently of whether each of the blocks is swapped or not, neither \((i, r+1)\) nor \((i + 1, r + 1)\) is a descent of \(U\) as well.

    The last (and more interesting) case is when \(\ind{a, c} + \ind{b, d} = 1\).
    This means that exactly one of \((i, r+1)\) and \((i + 1, r + 1)\) is a descent of \(T\).
    We also have \(\ind{a, d} + \ind{b, c} = 1\), so independently of whether each of the blocks is swapped or not, exactly one of \((i, r+1)\) and \((i + 1, r + 1)\) is a descent of \(U\) as well.
    If \(\ind{a, c} = \ind{a, d}\) (and consequently \(\ind{b, d} = \ind{b, c}\)), then the descent in \(T\) and the descent in \(U\) depend on the position of \(a\) and \(b\) (which is determined by whether \(B_1\) swaps or not), and are independent of the position of \(c\) and \(d\) (i.e., independent of whether \(B_2\) swaps or not).
    In this case, we \vocab{assign} the descent to the block \(B_1\).
    Otherwise, if \(\ind{a, c} = \ind{b, c}\) (and consequently \(\ind{b, d} = \ind{a, d}\)), then the descent in \(T\) and the descent in \(U\) depend on the position of \(c\) and \(d\) (which is determined by whether \(B_2\) swaps or not), and are independent of the position of \(a\) and \(b\) (i.e., independent of whether \(B_1\) swaps or not).
    In this case, we \vocab{assign} the descent to the block \(B_2\).

    All of these descents occurring at the interface of two blocks (or at row \(\alpha_{i+1} + 1\)) are called \vocab{inter-block descents}.

    Now, assume rows \(r\) and \(r+1\) are intertwined in \(T\), meaning that both rows belong to the same block \(B\), and that \(\ind{a, c} + \ind{b, d} + \ind{a, d} + \ind{b, c}\) is odd.
    Any descent occurring at box \((i, r+1)\) or \((i + 1, r+1)\) is called an \vocab{intra-block descent}, and we \vocab{assign} it to the block \(B\).

    If both or neither of \((i, r+1)\) and \((i + 1, r+1)\) are descents of \(T\), we say that row \(r+1\) is \vocab{balanced} with respect to \(T\).
    In terms of the indicators, this means that \(\ind{a, c} = \ind{b, d}\).
    Using the fact that \(\ind{a, c} + \ind{b, d} + \ind{a, d} + \ind{b, c}\) is odd,
    we can conlcude that \(\ind{a, b} = \ind{c, d}\) in this case, that is, \(\ind{r+1} = \ind{r}\).

    If only \((i, r+1)\) is a descent of \(T\), we say that row \(r+1\) is \vocab{unbalanced to the left} with respect to \(T\).
    In terms of the indicators, this means that \(\ind{a, c} = 1\) and \(\ind{b, d} = 0\).
    Using the fact that \(\ind{a, c} + \ind{b, d} + \ind{a, d} + \ind{b, c}\) is odd,
    we can conlcude that \(\ind{r+1} = \ind{a, b} = 1\) and \(\ind{r} = \ind{c, d} = 0\) in this case.

    If only \((i + 1, r+1)\) is a descent of \(T\), we say that row \(r+1\) is \vocab{unbalanced to the right} with respect to \(T\).
    In terms of the indicators, this means that \(\ind{a, c} = 0\) and \(\ind{b, d} = 1\).
    Using the fact that \(\ind{a, c} + \ind{b, d} + \ind{a, d} + \ind{b, c}\) is odd,
    we can conlcude that \(\ind{r+1} = \ind{a, b} = 0\) and \(\ind{r} = \ind{c, d} = 1\) in this case.

    Note that we can determine whether a row \(r+1\) is balanced, unbalanced to the left, or unbalanced to the right by looking only at \(\ind{r}\) and \(\ind{r+1}\): if \(\ind{r} = \ind{r+1}\), then row \(r+1\) is balanced; if \(\ind{r} = 0\) and \(\ind{r+1} = 1\), then row \(r+1\) is unbalanced to the left; and if \(\ind{r} = 1\) and \(\ind{r+1} = 0\), then row \(r+1\) is unbalanced to the right.

    We define the \vocab{major contributions} of a block \(B\) with respect to the fillings \(T\) and \(U\) as
    \begin{align*}
        \maj(B; T) &= \sum_{u \in \Des(T) \text{ assigned to } B} (\leg_{\alpha}(u) + 1) \\
        \maj(B; U) &= \sum_{u \in \Des(U) \text{ assigned to } B} (\leg_{s_i \permact \alpha}(u) + 1)
    \end{align*}
    We claim that for every block \(B\), we have \(\maj(B; T) = \maj(B; U)\).
    Let \(B\) be a block with signature entries \((a, b, c, d, e, f)\).
    Assume \(B\) is not the topmost block.

    Assume that \(B\) is swapped between \(T\) and \(U\).
    Then, if \(u\) is a descent assigned to \(B\) in \(T\), then \(s_i \cdot u\) is a descent assigned to \(B\) in \(U\), and vice versa, irregardless of whether the descent is inter-block or intra-block.
    Since \(\leg_{\alpha}(u) = \leg_{s_i \permact \alpha}(s_i \cdot u)\) for all boxes \(u\) in columns \(i\) or \(i+1\),
    we have \(\maj(B; T) = \maj(B; U)\) in this case.

    Now, assume that \(B\) is not swapped between \(T\) and \(U\).
    Let \(B = [r_1, r_2]\).
    This means that \(\ind{c, d, f} = \ind{e, a, b}\), that is,
    \begin{equation*}
        \ind{c, d} + \ind{d, f} - \ind{c, f} = \ind{e, a} + \ind{a, b} - \ind{e, b}.
    \end{equation*}
    First, assume \(\ind{a, b} = \ind{c, d}\), that is, \(\ind{r_2} = \ind{r_1}\).
    From our discussion about intra-block descents, in the sequence \((\ind{r_1}, \ind{r_1+1}, \dots, \ind{r_2})\), each time the indicator does not change, the corresponding row is balanced, and each time the indicator changes from \(0\) to \(1\) (resp. from \(1\) to \(0\)), the corresponding row is unbalanced to the left (resp. to the right).
    Since \(\ind{r_2} = \ind{r_1}\), the number of unbalanced to the left rows is equal to the number of unbalanced to the right rows in this block.
\end{proof}

Recall that the integral forms have a combinatorial formula in terms of augmented fillings using the signed alphabet \(\interval{n}_\pm\), as given in \Cref{eq:calE-Fill-signed}:
\begin{equation*}
    \mathcal{E}^{\sigma}_{\alpha}(\mathbf{x}; q, t) =
    \sum_{T \in \Fill[\interval{n}_\pm](\alpha, \sigma)}
    \mathbf{x}^{\content(T)}
    q^{\maj(T)} t^{\coinv(T)}
    (-t)^{\operatorname{neg}(T)}.
    %\tag{\ref{eq:calE-Fill-signed}}
\end{equation*}

Recall from \Cref{subsec:augmented-to-non-augmented} that any augmented filling \(T \in \Fill(\alpha, \sigma)\) may be identified with a non-augmented filling \(T' \in \Fill(\alpha + \mathbf{1})\) by moving each box one unit up.
This identification preserves \(\maj\), but changes \(\inv\) and \(\coinv\) by a constant depending only on \(\alpha\) and \(\sigma\).
We defined \(\symcoinv(T) = \coinv(T')\) and \(\syminv(T) = \inv(T')\).

We can apply \(\phi\) to \(T'\) to obtain \(U' = \phi(T') \in \Fill(s_i \permact (s_i \permact \alpha + \mathbf{1}))\), and then identify \(U'\) with an augmented filling \(U\) of shape \(\alpha\).
Note that \(\phi\) might or might not change the entries in positions \((i,1)\) and \((i+1,1)\) of \(T'\),
so the basement of \(U\) might be \(\sigma\) or \(\sigma s_i\).

Define the injection \(\phi \colon \Fill(\alpha, \sigma) \to \Fill(s_i \permact \alpha, \sigma) \cup \Fill(s_i \permact \alpha, \sigma s_i)\) by applying the bijection \(\phi\) to the non-augmented filling \(T'\) corresponding to \(T\), and then identifying the result with an augmented filling \(U\).
Similarly, define the injection \(\psi \colon \Fill(\alpha, \sigma) \to \Fill(s_i \permact \alpha, \sigma) \cup \Fill(s_i \permact \alpha, \sigma s_i)\) by applying the bijection \(\psi\) to the non-augmented filling \(T'\) corresponding to \(T\), and then identifying the result with an augmented filling \(U\).

We have immediate coloraries of \Cref{thm:phi-maj-inv} and \Cref{thm:psi-maj-inv}.
\begin{corollary} \label{cor:phi-symcoinv}
    For any \(T \in \Fill(\alpha, \sigma)\) and \(U = \phi(T)\), we have
    \begin{equation}
        \symcoinv(T) = \symcoinv(U)
        \quad \text{and} \quad
        \maj(T) = \maj(U).
    \end{equation}
\end{corollary}

\begin{corollary} \label{cor:psi-symcoinv}
    For any \(T \in \Fill(\alpha, \sigma)\) and \(U = \psi(T) \in \Fill(s_i \permact \alpha, \pi)\), we have
    \begin{itemize}
        \item if \(\ind{{\sigma_{i}},{\sigma_{i+1}}} = 1\) and \(\ind{{\pi_{i}},{\pi_{i+1}}} = 1\), then
        \begin{equation}
            \maj(T) = \maj(U) - (L+1)
            \quad \text{and} \quad
            \symcoinv(T) = \symcoinv(U) - A,
        \end{equation}
        \item if \(\ind{{\sigma_{i}},{\sigma_{i+1}}} = 1\) and \(\ind{{\pi_{i}},{\pi_{i+1}}} = 0\), then
        \begin{equation}
            \maj(T) = \maj(U)
            \quad \text{and} \quad
            \symcoinv(T) = \symcoinv(U) - 1,
        \end{equation}
        \item if \(\ind{{\sigma_{i}},{\sigma_{i+1}}} = 0\) and \(\ind{{\pi_{i}},{\pi_{i+1}}} = 1\), then
        \begin{equation}
            \maj(T) = \maj(U)
            \quad \text{and} \quad
            \symcoinv(T) = \symcoinv(U) + 1,
        \end{equation}
        \item and if \(\ind{{\sigma_{i}},{\sigma_{i+1}}} = 0\) and \(\ind{{\pi_{i}},{\pi_{i+1}}} = 0\), then
        \begin{equation}
            \maj(T) = \maj(U) + (L+1)
            \quad \text{and} \quad
            \symcoinv(T) = \symcoinv(U) + A,
        \end{equation}
    \end{itemize}
    where \(L = \leg(u)\) and \(A = \arm(u)\) for \(u = (i+1, \alpha_{i+1} + 1) \in \dg(s_i \permact \alpha)\).
\end{corollary}

\begin{definition}
    Define \(f \colon \Fill(\alpha, \sigma) \to \Fill(s_i \permact \alpha, \sigma)\)
    and \(g \colon \Fill(\alpha, \sigma) \to \Fill(s_i \permact \alpha, \sigma s_i)\) by
    \begin{equation}
        f(T) =
        \begin{cases}
            \phi(T), & \text{if } \phi(T) \in {\Fill(s_i \cdot \alpha, \sigma)}, \\
            \psi(T), & \text{if } \psi(T) \in {\Fill(s_i \cdot \alpha, \sigma)},
        \end{cases}
        \quad \text{and} \quad
        g(T) =
        \begin{cases}
            \phi(T), & \text{if } \phi(T) \in {\Fill(s_i \cdot \alpha, \sigma s_i)}, \\
            \psi(T), & \text{if } \psi(T) \in {\Fill(s_i \cdot \alpha, \sigma s_i)}.
        \end{cases}
    \end{equation}
\end{definition}

\begin{proposition}
    The maps \(f\) and \(g\) are well-defined and are bijections.
\end{proposition}

\begin{proof}
    Let \(T \in \Fill(\alpha, \sigma)\).
    Recall that the swapping conditions for \(\phi\) and \(\psi\) differ only in the bottommost block.
    Since the first row is always part of the bottommost block,
    exactly one of \(\phi(T)\) and \(\psi(T)\) has basement \(\sigma\),
    and the other has basement \(\sigma s_i\).
    Therefore, \(f\) and \(g\) are well-defined.

    The bijectivity of \(f\) and \(g\) follows from \Cref{prop:phi-inverse,prop:psi-inverse}.
\end{proof}

For the purposes of this section, let \(\wt(T) = \mathbf{x}^{\content(T)} q^{\maj(T)} t^{\coinv(T)} (-t)^{\operatorname{neg}(T)}\), for an augmented signed filling \(T\).

\begin{lemma} \label{lem:three-term-bijective-a}
    Assume that \(\alpha_i>\alpha_{i+1}\) and \(\sigma_i > \sigma_{i+1}\).
    Let \(T\) be an augmented filling in \(\Fill(\alpha, \sigma)\).
    Let \(U = f(T)\) and \(V = g(T)\).
    Then
    \begin{equation}
        \left( 1 - q^{L+1} t^{A + 1} \right) \wt(T)
        =
        \left( 1 - q^{L+1} t^{A} \right) \wt(U)
        +
        t^{\rho(\alpha,\sigma,i)} (1 - t) \wt(V).
    \end{equation}
\end{lemma}

\begin{proof}
    If \(U = \phi(T)\) and \(V = \psi(T)\),
    then \(\wt(U) = \wt(T)\) by \Cref{cor:phi-symcoinv},
    and \(\wt(V) = t^{-\rho(\alpha,\sigma,i)} q^{L+1} t^{A} \wt(T)\) by \Cref{cor:psi-symcoinv}.\todo{Explain these weight calculations more carefully.}
    Thus,
    \begin{multline*}
        \left( 1 - q^{L+1} t^{A} \right) \wt(U)
        +
        t^{\rho(\alpha,\sigma,i)} (1 - t) \wt(V)
        = \\
        \begin{aligned}
        &=
        \left( 1 - q^{L+1} t^{A} \right) \wt(T)
        +
        t^{\rho(\alpha,\sigma,i)} (1 - t) t^{-\rho(\alpha,\sigma,i)} q^{L+1} t^{A} \wt(T) \\
        &=
        \left( 1 - q^{L+1} t^{A} + (1 - t) q^{L+1} t^{A + 1} \right) \wt(T) \\
        &=
        \left( 1 - q^{L+1} t^{A + 1} \right) \wt(T).
        \end{aligned}
    \end{multline*}
    If \(U = \psi(T)\) and \(V = \phi(T)\),
    then \(\wt(U) = t \wt(T)\) by \Cref{cor:psi-symcoinv},
    and \(\wt(V) = t^{-\rho(\alpha,\sigma,i)} \wt(T)\) by \Cref{cor:phi-symcoinv}.
    Thus,
    \begin{multline*}
        \left( 1 - q^{L+1} t^{A} \right) \wt(U)
        +
        t^{\rho(\alpha,\sigma,i)} (1 - t) \wt(V)
        = \\
        \begin{aligned}
        &=
        \left( 1 - q^{L+1} t^{A} \right) t \wt(T)
        +
        t^{\rho(\alpha,\sigma,i)} (1 - t) t^{-\rho(\alpha,\sigma,i)} \wt(T) \\
        &=
        \left( t - q^{L+1} t^{A + 1} + 1 - t \right) \wt(T) \\
        &=
        \left( 1 - q^{L+1} t^{A + 1} \right) \wt(T).
        \end{aligned}
    \end{multline*}
    Therefore, in either case, the equation holds.
\end{proof}

\begin{corollary} \label{cor:three-term-a}
    Let \(\alpha\) be a composition such that \(\alpha_i > \alpha_{i+1}\) and \(\sigma \in S_n\) such that \(\sigma_i > \sigma_{i+1}\).
    Then,
    \begin{equation} \label{eq:three-term-a-integral}
        \left( 1 - q^{L+1} t^{A + 1} \right) \mathcal{E}_{\alpha}^\sigma
        = 
        \left( 1 - q^{L+1} t^{A} \right) \mathcal{E}_{s_i \permact \alpha}^{\sigma s_i} 
        +
        t^{\rho(\alpha,\sigma,i)} (1 - t) \mathcal{E}_{s_i \permact \alpha}^\sigma,
    \end{equation}
    where \(A=\arm(u)\) and \(L=\leg(u)\) for \(u=(i+1,\alpha_{i+1} +1)\in\dg(s_i \permact \alpha)\), and \( \rho(\alpha,\sigma,i)=\twinv(s_i \permact \alpha,\sigma)-\twinv(\alpha,\sigma)\).
\end{corollary}

\begin{proof}
    This follows by summing the equation in \Cref{lem:three-term-bijective-a} over all \(T \in \Fill(\alpha, \sigma)\).
\end{proof}

\begin{theorem} \label{thm:three-term-a-non-integral}
    Let \(\alpha\) be a composition such that \(\alpha_i > \alpha_{i+1}\) and \(\sigma \in S_n\) such that \(\sigma_i > \sigma_{i+1}\).
    Then,
    \begin{equation}
        E_{\alpha}^\sigma(X;q;t)=E_{s_i \permact \alpha}^{\sigma s_i}(X;q;t)+t^{\rho(\alpha,\sigma,i)}\frac{1-t}{1-q^{L+1}t^{A}}E_{s_i \permact \alpha}^\sigma(X;q;t)
    \end{equation}
    where \(A=\arm(u)\) and \(L=\leg(u)\) for \(u=(i+1,\alpha_{i+1} +1)\in\dg(s_i \permact \alpha)\), and  \( \rho(\alpha,\sigma,i)=\twinv(s_i \permact \alpha,\sigma)-\twinv(\alpha,\sigma)\).
\end{theorem}

\begin{proof}
    This follows from \Cref{lem:three-term-bijective-a} by dividing both sides of \Cref{eq:three-term-a-integral} by the appropriate factors to convert the integral forms to non-integral forms.
\end{proof}

\begin{corollary} \label{cor:three-term-b}
    Let \(i \in \interval{n-1}\),
    \(\alpha = \composition{\alpha_1, \dots, \alpha_n}\) be a composition with \(\alpha_i < \alpha_{i+1}\),

\end{theorem}
